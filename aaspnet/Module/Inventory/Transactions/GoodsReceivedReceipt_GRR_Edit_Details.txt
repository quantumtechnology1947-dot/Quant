    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            GRRNo = Request.QueryString["GRRNo"].ToString();
            GINNo = Request.QueryString["GINNo"].ToString();
            SupplierNo = Request.QueryString["SupId"].ToString();
            poNo = Request.QueryString["PONo"].ToString();

            sId = Session["username"].ToString();
            FinYearId = Convert.ToInt32(Session["finyear"]);
            CompId = Convert.ToInt32(Session["compid"]);
            MId = Request.QueryString["Id"].ToString();
            GINId = Request.QueryString["GINId"].ToString();

            connStr = fun.Connection();
            con = new SqlConnection(connStr);
            con.Open();

            if (!Page.IsPostBack)
            {
                lblGrr.Text = GRRNo;
                lblGIn.Text = GINNo;

                string sqlSup = fun.select("SupplierName", "tblMM_Supplier_master", "CompId='" + CompId + "' AND SupplierId='" + SupplierNo + "'");

                SqlCommand cmdSupId = new SqlCommand(sqlSup, con);
                SqlDataAdapter daSupId = new SqlDataAdapter(cmdSupId);
                DataSet DSSupId = new DataSet();
                daSupId.Fill(DSSupId);

                if (DSSupId.Tables[0].Rows.Count > 0)
                {
                    lblSupplier.Text = DSSupId.Tables[0].Rows[0][0].ToString();
                }
                string sql = fun.select("*", "tblInv_Inward_Master", "Id='" + GINId + "' AND CompId='" + CompId + "'");
                SqlCommand cmd = new SqlCommand(sql, con);
                SqlDataAdapter da = new SqlDataAdapter(cmd);
                DataSet DS = new DataSet();
                da.Fill(DS);

                if (DS.Tables[0].Rows.Count > 0)
                {
                    lblChNo.Text = DS.Tables[0].Rows[0]["ChallanNo"].ToString();
                    lblDate.Text = fun.FromDateDMY(DS.Tables[0].Rows[0]["ChallanDate"].ToString());
                }
                this.loadData();
                this.disableEdit();
            }
        }
        catch (Exception ex) { }
        finally
        { con.Close(); }
    }


    public void disableEdit()
    {
        foreach (GridViewRow row in GridView2.Rows)
        {
            int Id = Convert.ToInt32(((Label)row.FindControl("lblId")).Text);

            string SqlStr = fun.select("tblinv_MaterialReceived_Details.Id", "tblinv_MaterialReceived_Master,tblQc_MaterialQuality_Details,tblQc_MaterialQuality_Master,tblinv_MaterialReceived_Details", "tblQc_MaterialQuality_Master.Id=tblQc_MaterialQuality_Details.MId AND tblinv_MaterialReceived_Master.Id=tblinv_MaterialReceived_Details.MId AND tblinv_MaterialReceived_Master.CompId ='" + CompId + "' AND tblinv_MaterialReceived_Master.Id=tblQc_MaterialQuality_Master.GRRId AND tblinv_MaterialReceived_Master.Id='" + Id + "' AND tblinv_MaterialReceived_Details.Id=tblQc_MaterialQuality_Details.GRRId");

            SqlCommand SQLCmd = new SqlCommand(SqlStr, con);
            SqlDataAdapter SqlDa = new SqlDataAdapter(SQLCmd);
            DataSet SqlDs = new DataSet();

            SqlDa.Fill(SqlDs);

            if (SqlDs.Tables[0].Rows.Count > 0)
            {
                ((LinkButton)row.FindControl("LinkButton1")).Visible = false;
                ((Label)row.FindControl("lblgqn")).Visible = true;
            }
            else
            {
                ((LinkButton)row.FindControl("LinkButton1")).Visible = true;
                ((Label)row.FindControl("lblgqn")).Visible = false;
            }
        }
    }

                            dr[4] = decimal.Parse((DSSql3.Tables[0].Rows[0]["Qty"]).ToString()).ToString("N3");

                        }
                    }

                    if (DSSql2.Tables[0].Rows[0]["ReceivedQty"].ToString() == "")
                    {
                        dr[5] = "0";
                    }
                    else
                    {
                        dr[5] = decimal.Parse((DSSql2.Tables[0].Rows[0]["ReceivedQty"]).ToString()).ToString("N3");
                    }
                }

                if (DSSql.Tables[0].Rows[i]["ReceivedQty"].ToString() == "")
                {
                    dr[6] = "0";
                }
                else
                {
                    dr[6] = decimal.Parse((DSSql.Tables[0].Rows[i]["ReceivedQty"]).ToString()).ToString("N3");
                }
                POId = Convert.ToInt32(DSSql.Tables[0].Rows[i]["POId"]);

                dr[7] = POId;

                string sqlget = fun.select("sum(ReceivedQty) as sum_ReceivedQty", "tblinv_MaterialReceived_Master,tblinv_MaterialReceived_Details", "tblinv_MaterialReceived_Master.CompId='" + CompId + "' AND tblinv_MaterialReceived_Master.GRRNo=tblinv_MaterialReceived_Details.GRRNo AND tblinv_MaterialReceived_Details.POId='" + DSSql.Tables[0].Rows[i]["POId"].ToString() + "' AND tblinv_MaterialReceived_Master.Id=tblinv_MaterialReceived_Details.MId AND tblinv_MaterialReceived_Master.GINId='" + GINId + "'");

                SqlCommand cmdget = new SqlCommand(sqlget, con);
                SqlDataAdapter daget = new SqlDataAdapter(cmdget);
                DataSet DSget = new DataSet();
                daget.Fill(DSget);

                if (DSget.Tables[0].Rows[0]["sum_ReceivedQty"] != DBNull.Value)
                {
                    dr[8] = decimal.Parse((DSget.Tables[0].Rows[0]["sum_ReceivedQty"]).ToString()).ToString("N3");
                }
                else
                {
                    dr[8] = "0";
                }


                dr[11]=ItemId;
                dt.Rows.Add(dr);
                dt.AcceptChanges();
            }

            GridView2.DataSource = dt;
            GridView2.DataBind();

        }
        catch (Exception ex) { }
        finally

        { con.Close(); }

    }

    protected void GridView2_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            con.Open();

            string CDate = fun.getCurrDate();
            string CTime = fun.getCurrTime();
            int index = GridView2.EditIndex;
            GridViewRow row = GridView2.Rows[index];
            int id = Convert.ToInt32(GridView2.DataKeys[e.RowIndex].Value);
            double RecvQty = 0;
            double lblRecvQty = 0;
            double InwrdQty = 0;
            RecvQty = Convert.ToDouble(decimal.Parse((((TextBox)row.FindControl("TxtRecedQty")).Text).ToString()).ToString("N3"));
            lblRecvQty = Convert.ToDouble(decimal.Parse((((Label)row.FindControl("lblRecedqty1")).Text).ToString()).ToString("N3"));
            InwrdQty = Convert.ToDouble(decimal.Parse((((Label)row.FindControl("lblInwrdqty")).Text).ToString()).ToString("N3"));
            if (fun.NumberValidationQty(RecvQty.ToString()) == true && RecvQty.ToString() != "")
            {
                if (RecvQty <= lblRecvQty)
                {

                    string upd = fun.update("tblinv_MaterialReceived_Master", "SysDate='" + CDate + "' ,SysTime='" + CTime + "',SessionId='" + sId + "'", "Id='" + MId + "' AND CompId='" + CompId + "'");
                    SqlCommand cmd = new SqlCommand(upd, con);
                    cmd.ExecuteNonQuery();
                    string upd1 = fun.update("tblinv_MaterialReceived_Details", "ReceivedQty='" + RecvQty + "'", "Id='" + id + "'");
                    SqlCommand cmd2 = new SqlCommand(upd1, con);
                    cmd2.ExecuteNonQuery();
                    GridView2.EditIndex = -1;
                    this.loadData();

                }
                else
                {
                    string mystringmsg = string.Empty;
                    mystringmsg = " Received Qty exceed limit.";
                    ClientScript.RegisterStartupScript(this.GetType(), "alert", "alert('" + mystringmsg + "');", true);
                }
            }
        }
        catch (Exception ex) { }
        finally

        { con.Close(); }
    }
