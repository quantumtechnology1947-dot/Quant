    [System.Web.Services.WebMethodAttribute(), System.Web.Script.Services.ScriptMethodAttribute()]
    public static string[] sql(string prefixText, int count, string contextKey)
    {
        clsFunctions fun = new clsFunctions();
        string connStr = fun.Connection();
        DataSet ds = new DataSet();
        SqlConnection con = new SqlConnection(connStr);
        int CompId = Convert.ToInt32(HttpContext.Current.Session["compid"]);
        con.Open();
        string cmdStr = fun.select("SupplierId,SupplierName", "tblMM_Supplier_master","CompId='"+CompId+"'");
        SqlDataAdapter da = new SqlDataAdapter(cmdStr, con);
        da.Fill(ds, "tblMM_Supplier_master");
        con.Close();
        string[] main = new string[0];
        for (int i = 0; i < ds.Tables[0].Rows.Count; i++)
        {
            if (ds.Tables[0].Rows[i].ItemArray[1].ToString().ToLower().StartsWith(prefixText.ToLower()))
            {
                Array.Resize(ref main, main.Length + 1);
                main[main.Length - 1] = ds.Tables[0].Rows[i].ItemArray[1].ToString() + " [" + ds.Tables[0].Rows[i].ItemArray[0].ToString() + "]";
                //if (main.Length == 10)
                //    break;
            }
        }
        Array.Sort(main);
        return main;
    }

    public void loadData( string SupId, string Po)
    {
        try
        {

            
            con.Open();

            string y = "";
            string x = "";
            string sid = "";

            if (DropDownList1.SelectedValue == "1")
            {
                y = " AND tblInv_Inward_Master.PONo='" + txtEnqId.Text + "'";
            }
            if (DropDownList1.SelectedValue == "2")
            {
                y = " AND tblInv_Inward_Master.GINNo='" + txtEnqId.Text + "'";
            }
            if (DropDownList1.SelectedValue == "0")
            {
                if (txtSupplier.Text != "")
                {
                    sid = fun.getCode(txtSupplier.Text);
                    x = " AND tblMM_PO_Master.SupplierId='" + sid + "'";
                }
            }
            SqlDataAdapter da = new SqlDataAdapter("Sp_GIN_Edit", con);
            da.SelectCommand.CommandType = CommandType.StoredProcedure;
            da.SelectCommand.Parameters.Add(new SqlParameter("@CompId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@CompId"].Value = CompId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@FinId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@FinId"].Value = FyId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@x", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@x"].Value = x;
            da.SelectCommand.Parameters.Add(new SqlParameter("@y", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@y"].Value = y;
            DataSet DSitem = new DataSet();
            da.Fill(DSitem);
            GridView2.DataSource = DSitem;
            GridView2.DataBind();
            this.disableEdit();
           
        }
        catch (Exception ex) { }
        finally
        { 
            con.Close();
        }
    }

    protected void GridView2_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            int index = GridView2.EditIndex;
            GridViewRow row = GridView2.Rows[index];
            int id = Convert.ToInt32(GridView2.DataKeys[e.RowIndex].Value);
            string ChNo = ((TextBox)row.FindControl("TxtChNo")).Text;
            string ChDt = fun.FromDate(((TextBox)row.FindControl("TxtChDt")).Text);

            if (ChNo != "0" && ChNo != "" && ChDt != "")
            {

                string sql = fun.update("tblInv_Inward_Master", "ChallanNo='" + ChNo + "', ChallanDate='" + ChDt + "'", "Id=" + id + " AND CompId='" + CompId + "'");
                SqlCommand cmd = new SqlCommand(sql, con);
                con.Open();
                int temp = cmd.ExecuteNonQuery();
                con.Close();
                if (temp == 1)
                {

                    string mystring = string.Empty;
                    mystring = "Record is updated successfully.";
                    ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
                }
                GridView2.EditIndex = -1;
                this.loadData(j,m);

            }

            else
            {
                string mystring = string.Empty;
                mystring = "Please enter Challan No Or Challan Date.";
                ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
            }
        }
        catch (Exception ex) { }

    }
