﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_PopUpNews : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();

    protected void Page_Load(object sender, EventArgs e)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);

        try
        {
            con.Open();
            int id = 0;
            int CId = Convert.ToInt32(Session["compid"]);

            if (Request.QueryString["Id"] != null && Request.QueryString["Id"] != "")
            {
                id = Convert.ToInt32(Request.QueryString["Id"]);

                string cmdgrid = fun.select("*", "tblHR_News_Notices", "Id='" + id + "'");
                SqlCommand cmd = new SqlCommand(cmdgrid, con);
                SqlDataAdapter da = new SqlDataAdapter(cmd);
                DataSet ds = new DataSet();
                da.Fill(ds);

                lbldate.Text = fun.FromDateDMY(ds.Tables[0].Rows[0]["SysDate"].ToString());
                lblTime.Text = ds.Tables[0].Rows[0]["SysTime"].ToString();
                lblMsg.Text = ds.Tables[0].Rows[0]["InDetails"].ToString();
                lblTitle.Text = ds.Tables[0].Rows[0]["Title"].ToString();

            }

            con.Close();

        }
        catch (Exception ex)
        {

        }
    }


    protected void Button1_Click(object sender, EventArgs e)
    {
        Response.Write("<script language=javascript> window.close()</script>");
        Response.End();
    }


}
