﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_Design_Masters_CategoryEdit : System.Web.UI.Page
{

    clsFunctions fun = new clsFunctions();

    string compid = "";
    string finyrsid = "";
    string sessionid = "";
    string connStr = "";
    SqlConnection con;
    int s = 0;
    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            lblMessage.Text = "";
            compid = Session["compid"].ToString();
            finyrsid = Session["finyear"].ToString();
            sessionid = Session["username"].ToString();
            connStr = fun.Connection();
            con = new SqlConnection(connStr);

        }

        catch (Exception ex)
        {
        }

    }

    protected void GridView1_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        if (e.Row.RowType == DataControlRowType.DataRow)
        {
            LinkButton upd = (LinkButton)e.Row.Cells[1].Controls[0];
            upd.Attributes.Add("onclick", "return confirmationUpdate();");
        }

    }
    protected void GridView1_RowUpdated(object sender, GridViewUpdatedEventArgs e)
    {
        if (s > 0)
        {
            lblMessage.Text = "Record Updated.";
        }
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {

        try
        {
            if (e.CommandName == "Update")
            {
                int rowIndex = Convert.ToInt32(e.CommandArgument);
                GridViewRow row = GridView1.Rows[rowIndex];
                int CId = 0;
                CId = Convert.ToInt32(((Label)row.FindControl("Label1")).Text);
                string strCname = ((TextBox)row.FindControl("txtCate")).Text;
                LocalSqlServer.UpdateParameters["CId2"].DefaultValue = CId.ToString();
                LocalSqlServer.UpdateParameters["CName2"].DefaultValue = strCname.ToString();
                LocalSqlServer.Update();
                s++;
            }

        }

        catch (Exception ex)
        { }
    }
}
