    public void LoadData()
    {
       try
        {
            con.Open(); 
            SqlDataAdapter da = new SqlDataAdapter("[Sp_MRN_FillGrid_EDPD]", con);
            da.SelectCommand.CommandType = CommandType.StoredProcedure;
            da.SelectCommand.Parameters.Add(new SqlParameter("@CompId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@CompId"].Value = CompId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@FinId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@FinId"].Value = FyId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@x", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@x"].Value = MId;            
            DataSet DSitem = new DataSet();
            da.Fill(DSitem);
            GridView1.DataSource = DSitem;
            GridView1.DataBind();            

        }
       catch (Exception ex) { }
       finally
        {
            con.Close();
        }
    }

    protected void GridView1_RowEditing(object sender, GridViewEditEventArgs e)
    {
        try
        {
            GridView1.EditIndex = e.NewEditIndex;
            this.LoadData();
            int index = GridView1.EditIndex;
            GridViewRow grv = GridView1.Rows[index];
            ((DropDownList)grv.FindControl("DropDownList1")).Visible = true;
            ((Label)grv.FindControl("lblWODept")).Visible = false;
            string DeptId = ((Label)grv.FindControl("lblDwpt")).Text;
            if (DeptId == "1" && ((Label)grv.FindControl("lblWODept")).Text == "WONo")
            {                
                ((DropDownList)grv.FindControl("DropDownList1")).SelectedValue = "2";
                if (((DropDownList)grv.FindControl("DropDownList1")).SelectedValue == "2")
                {
                    ((TextBox)grv.FindControl("txtwono")).Visible = true;
                    if (((TextBox)grv.FindControl("txtwono")).Visible == true)
                    {
                        ((RequiredFieldValidator)grv.FindControl("Req1")).Visible = true;
                    }
                    ((Label)grv.FindControl("lblDW")).Visible = false;
                }
                else
                {
                    ((TextBox)grv.FindControl("txtwono")).Visible = false;
                    if (((TextBox)grv.FindControl("txtwono")).Visible == false)
                    {
                        ((RequiredFieldValidator)grv.FindControl("Req1")).Visible = false;
                    }

                }
            }
            else
            {
                ((DropDownList)grv.FindControl("DropDownList1")).SelectedValue = "1";

                if (((DropDownList)grv.FindControl("DropDownList1")).SelectedValue == "1")
                {
                    ((DropDownList)grv.FindControl("drpdept")).Visible = true;
                    ((Label)grv.FindControl("lblDW")).Visible = false;
                }
                else
                {
                    ((DropDownList)grv.FindControl("drpdept")).Visible = false;

                }
            }
        }
        catch (Exception ex)
        {
        }
    }
