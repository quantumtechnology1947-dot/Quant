    public void fillGrid(string sql, GridView GridView2)
    {
        string connStr = fun.Connection();
        DataSet DS3 = new DataSet();
        SqlConnection con = new SqlConnection(connStr);
        try
        {
            SqlCommand cmd = new SqlCommand(sql, con);
            SqlDataAdapter DA2 = new SqlDataAdapter(cmd);
            DA2.Fill(DS3);           

            DataTable dt = new DataTable();

            dt.Columns.Add("Id", typeof(string));
            dt.Columns.Add("ItemCode", typeof(string));
            dt.Columns.Add("ManfDesc", typeof(string));           
            dt.Columns.Add("UOMBasic", typeof(string));          
            dt.Columns.Add("Location", typeof(string));
            dt.Columns.Add("Qty", typeof(string));
            dt.Columns.Add("MId", typeof(string));
            dt.Columns.Add("FileName", typeof(string));

            DataRow dr;

            for (int k = 0; k < DS3.Tables[0].Rows.Count; k++)
            {

                dr = dt.NewRow();
                dr[0] = DS3.Tables[0].Rows[k]["Id"].ToString();
                dr[1] = DS3.Tables[0].Rows[k]["ItemCode"].ToString();
                dr[2] = DS3.Tables[0].Rows[k]["ManfDesc"].ToString();
              

                string sql10 = fun.select("Symbol ", "Unit_Master", "Id='" + DS3.Tables[0].Rows[k]["UOMBasic"].ToString() + "'");

                SqlCommand cmd10 = new SqlCommand(sql10, con);
                SqlDataAdapter DA10 = new SqlDataAdapter(cmd10);
                DataSet DS10 = new DataSet();
                DA10.Fill(DS10);
                if (DS10.Tables[0].Rows.Count>0)
                {
                dr[3] = DS10.Tables[0].Rows[0]["Symbol"].ToString();
                }

                

                string sql2 = fun.select("Location,FileName,Id", " tblMS_Master ", " ItemId='" + DS3.Tables[0].Rows[k]["Id"] + "'");

                    SqlCommand cmd2 = new SqlCommand(sql2, con);
                    SqlDataAdapter DA3 = new SqlDataAdapter(cmd2);
                    DataSet DS4 = new DataSet();
                    DA3.Fill(DS4);

                    if (DS4.Tables[0].Rows.Count > 0)
                    {

                        dr[4] = DS4.Tables[0].Rows[0]["Location"].ToString();
                        dr[7] = DS4.Tables[0].Rows[0]["FileName"].ToString();
                        dr[6] = DS4.Tables[0].Rows[0]["Id"].ToString();
                    }
                   

                    dr[5] = DS3.Tables[0].Rows[k]["StockQty"].ToString(); ;
                

                dt.Rows.Add(dr);
                dt.AcceptChanges();
            }

            GridView2.DataSource = dt;
            GridView2.DataBind();
        }
        catch (Exception ex)
        {

        }
       finally
        {
            con.Close();
        }
    }
    protected void GridView2_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {

            GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
            int id = 0;
            id = Convert.ToInt32(((Label)row.FindControl("lblId")).Text);
            int mid = Convert.ToInt32(((Label)row.FindControl("lblMId")).Text); 

            if (e.CommandName == "Sel")
            {                
                if(id!=0)
                {
                    string getRandomKey = fun.GetRandomAlphaNumeric();

                    Response.Redirect("~/Module/Machinery/Masters/Machinery_Print_Details.aspx?Id=" + id + "&Key=" + getRandomKey + "&ModId=15&SubModId=67");  
                }
            }


            if (e.CommandName == "Download")
            {
                string StrRef = fun.select("*", "tblMS_Master", "CompId='" + CompId + "' AND Id='" + mid + "'");

                SqlCommand cmdRef = new SqlCommand(StrRef, con);
                SqlDataAdapter DARef = new SqlDataAdapter(cmdRef);
                DataSet DSRef = new DataSet();
                DARef.Fill(DSRef);

                if (DSRef.Tables[0].Rows[0]["FileName"] == DBNull.Value || DSRef.Tables[0].Rows[0]["FileName"].ToString() == "")
                {

                    ((LinkButton)row.FindControl("LinkBtnDownload")).Visible = false;
                }
                else
                {

                    ((LinkButton)row.FindControl("LinkBtnDownload")).Visible = true;
                    Response.Redirect("~/Controls/DownloadFile.aspx?Id=" + DSRef.Tables[0].Rows[0]["Id"].ToString() + "&tbl=tblMS_Master&qfd=FileData&qfn=fileName&qct=ContentType");
                }
            }

        }
        catch (Exception es)
        {
        }

    }
