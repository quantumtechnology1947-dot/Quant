﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;

public partial class Module_SalesDistribution_Masters_CategoryNew : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    string sysdate = "";
    string systime = "";
    string compid = "";
    string finyrsid = "";
    string sessionid = "";

    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            lblMessage.Text = "";
            sysdate = fun.getCurrDate();
            systime = fun.getCurrTime();
            compid = Session["compid"].ToString();
            finyrsid = Session["finyear"].ToString();
            sessionid = Session["username"].ToString();
            if (!IsPostBack)
            {
                this.fillGrid();
            }
        }
       catch (Exception ex)
        {
        }
    }

    public void fillGrid()
    {
       try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            con.Open();
            DataTable dt = new DataTable();

            dt.Columns.Add(new System.Data.DataColumn("CId", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("CName", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("HasSubCat", typeof(string)));

            string sql = fun.select("*", "tblSD_WO_Category", "CompId='" + compid + "' AND FinYearId<='" + finyrsid + "' Order by CId desc");

            SqlCommand cmd = new SqlCommand(sql, con);
            SqlDataAdapter da = new SqlDataAdapter(cmd);
            DataSet ds = new DataSet();
            da.Fill(ds);

            DataRow dr;

            for (int i = 0; i < ds.Tables[0].Rows.Count; i++)
            {
                dr = dt.NewRow();
                dr[0] = Convert.ToInt32(ds.Tables[0].Rows[i]["CId"].ToString());
                dr[1] = ds.Tables[0].Rows[i]["CName"].ToString();
                dr[2] = ds.Tables[0].Rows[i]["Symbol"].ToString();
                if (ds.Tables[0].Rows[i]["HasSubCat"].ToString() == "1")
                {
                    dr[3] = "Yes";
                }
                else
                {
                    dr[3] = "No";
                }

                dt.Rows.Add(dr);
                dt.AcceptChanges();


            }

            GridView1.DataSource = dt;
            GridView1.DataBind();
            con.Close();
        }

      catch (Exception ex)
        {
        }

    }


    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {
     try
        {

            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            if (e.CommandName == "Add")
            {
                string strUnitName = ((TextBox)GridView1.FooterRow.FindControl("txtCName")).Text;
                string strAbbrivation = ((TextBox)GridView1.FooterRow.FindControl("txtAbb")).Text;
                string strHasSubCat = "";

                if (strUnitName != "" && strAbbrivation != "")
                {

                    if (((CheckBox)GridView1.FooterRow.FindControl("CheckBox1")).Checked == true)
                    {
                        strHasSubCat = "1";
                    }
                    else
                    {
                        strHasSubCat = "0";
                    }


                    string sql = fun.select("Symbol", "tblSD_WO_Category", "CompId='" + compid + "' And Symbol='" + strAbbrivation + "' AND FinYearId<='" + finyrsid + "' Order by CId desc");
                    SqlCommand cmd1 = new SqlCommand(sql, con);
                    SqlDataAdapter da = new SqlDataAdapter(cmd1);
                    DataSet ds = new DataSet();
                    da.Fill(ds);
                    if (ds.Tables[0].Rows.Count == 0)
                    {

                        string InsertCommand = fun.insert("tblSD_WO_Category", "SysDate,SysTime,CompId,FinYearId,SessionId,CName, Symbol,HasSubCat", "'" + sysdate + "','" + systime + "','" + compid + "','" + finyrsid + "','" + sessionid + "','" + strUnitName + "','" + strAbbrivation.ToUpper() + "','" + strHasSubCat + "'");
                        SqlCommand cmd = new SqlCommand(InsertCommand, con);
                        con.Open();
                        cmd.ExecuteNonQuery();
                        con.Close();
                        Page.Response.Redirect(Request.Url.ToString(), true);
                    }
                    else
                    {

                        string mystring = string.Empty;
                        mystring = "Category symbol is already used.";
                        ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
                    }
                }

            }


            if (e.CommandName == "Add1")
            {
                string strHasSubCat = "";
        
                string strUnitName = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtCName")).Text;
                string strAbbrivation = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtAbb")).Text;

                if (strUnitName != "" && strAbbrivation != "")
                {

                    if (((CheckBox)GridView1.Controls[0].Controls[0].FindControl("CheckBox1")).Checked == true)
                    {
                        strHasSubCat = "1";
                    }
                    else
                    {
                        strHasSubCat = "0";
                    }


                    string sql = fun.select("Symbol", "tblSD_WO_Category", "CompId='" + compid + "' And Symbol='" + strAbbrivation + "' AND FinYearId<='" + finyrsid + "' Order by CId desc");
                    SqlCommand cmd1 = new SqlCommand(sql, con);
                    SqlDataAdapter da = new SqlDataAdapter(cmd1);
                    DataSet ds = new DataSet();
                    da.Fill(ds);
                    if (ds.Tables[0].Rows.Count == 0)
                    {
                        string InsertCommand = fun.insert("tblSD_WO_Category", "SysDate,SysTime,CompId,FinYearId,SessionId,CName, Symbol,HasSubCat", "'" + sysdate + "','" + systime + "','" + compid + "','" + finyrsid + "','" + sessionid + "','" + strUnitName + "','" + strAbbrivation.ToUpper() + "','" + strHasSubCat + "'");
                        SqlCommand cmd = new SqlCommand(InsertCommand, con);
                        con.Open();
                        cmd.ExecuteNonQuery();
                        con.Close();
                        Page.Response.Redirect(Request.Url.ToString(), true);

                    }
                    else
                    {

                        string mystring = string.Empty;
                        mystring = "Category symbol is already used.";
                        ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
                    }
                }

            }
        }

      catch (Exception ex)
        {
        }
    }

    }
