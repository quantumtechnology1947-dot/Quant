﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;

public partial class Module_HR_Masters_PF_Slab : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    string connStr = "";
    SqlConnection con;
    protected void Page_Load(object sender, EventArgs e)
    {
       
        lblMessage.Text = "";        
        connStr = fun.Connection();
        con = new SqlConnection(connStr);
        if(!IsPostBack)
        {
        this.FillAccegrid();
        }

    }
    protected void GridView1_RowUpdated(object sender, GridViewUpdatedEventArgs e)
    {
        lblMessage.Text = "Record Updated.";
    }
    protected void GridView1_RowDeleted(object sender, GridViewDeletedEventArgs e)
    {
        lblMessage.Text = "Record Deleted.";
    }


    public void FillAccegrid()
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            DataTable dt = new DataTable();
            string sql = "SELECT * FROM [tblHR_PF_Slab] order by [Id] desc";
            SqlCommand cmd = new SqlCommand(sql, con);
            SqlDataAdapter da = new SqlDataAdapter(cmd);
            DataSet ds = new DataSet();
            da.Fill(ds);
            GridView1.DataSource = ds;
            GridView1.DataBind();
            string sql2 = "SELECT Active FROM [tblHR_PF_Slab] where Active=1 order by [Id] desc";
            SqlCommand cmd2 = new SqlCommand(sql2, con);
            SqlDataAdapter da2 = new SqlDataAdapter(cmd2);
            DataSet ds2 = new DataSet();
            da2.Fill(ds2);
            if(ds2.Tables[0].Rows.Count>0)
            {
                ((CheckBox)GridView1.FooterRow.FindControl("ChkActive")).Enabled= false;
            }           
        }
        catch (Exception ex) { }
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {

        try
        {
            if (e.CommandName == "Add")
            {
                string strPFEmployee = ((TextBox)GridView1.FooterRow.FindControl("txtPFEmployee")).Text;
                string strPFCompany = ((TextBox)GridView1.FooterRow.FindControl("txtPFCompany")).Text;

                int strActive = 0;

                if (((CheckBox)GridView1.FooterRow.FindControl("ChkActive")).Checked == true)
                {
                    strActive = 1;
                }


                if (strPFEmployee != "" && strPFCompany != "" && fun.NumberValidation(((TextBox)GridView1.FooterRow.FindControl("txtPFEmployee")).Text) == true && fun.NumberValidation(((TextBox)GridView1.FooterRow.FindControl("txtPFCompany")).Text) == true)
                {

                    string insert1 = fun.insert("tblHR_PF_Slab", "PFEmployee,PFCompany,Active ", "'" + strPFEmployee + "','" + strPFCompany + "','" + strActive + "'");

                    SqlCommand cmd1 = new SqlCommand(insert1, con);
                    con.Open();
                    cmd1.ExecuteNonQuery();
                    con.Close();
                    this.FillAccegrid();

                }

            }
            else if (e.CommandName == "Add1")
            {
                string strPFEmployee = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtPFEmployee1")).Text;
                string strPFCompany = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtPFCompany1")).Text;

                int strActive = 0;

                if (((CheckBox)GridView1.Controls[0].Controls[0].FindControl("ChkActive1")).Checked == true)
                {
                    strActive = 1;
                }
                if (strPFEmployee != "" && strPFCompany != "" && fun.NumberValidation(((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtPFEmployee1")).Text) == true && fun.NumberValidation(((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtPFCompany1")).Text) == true)
                {
                    string insert1 = fun.insert("tblHR_PF_Slab", "PFEmployee,PFCompany,Active ", "'" + strPFEmployee + "','" + strPFCompany + "','" + strActive + "'");

                    SqlCommand cmd1 = new SqlCommand(insert1, con);
                    con.Open();
                    cmd1.ExecuteNonQuery();
                    con.Close();
                    this.FillAccegrid();
                }
            }
        }

        catch (Exception ex)
        {
        }
    }

    protected void GridView1_RowDataBound(object sender, GridViewRowEventArgs e)
    {
         try
        {
            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton edit = (LinkButton)e.Row.Cells[1].Controls[0];
                edit.Attributes.Add("onclick", "return confirmationUpdate();");
            }

            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del = (LinkButton)e.Row.Cells[2].Controls[0];
                del.Attributes.Add("onclick", "return confirmationDelete();");
            }
        }
       catch(Exception ex){}

    }

    protected void GridView1_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            int rowIndex = GridView1.EditIndex;
            GridViewRow row = GridView1.Rows[rowIndex];
            int id = Convert.ToInt32(GridView1.DataKeys[e.RowIndex].Value);

            string strPFEmployee = ((TextBox)row.FindControl("txtPFEmployee0")).Text;
            string strPFCompany = ((TextBox)row.FindControl("txtPFCompany0")).Text;
            int strActive = 0;

            if (((CheckBox)row.FindControl("ChkActive0")).Checked == true)
            {
                strActive = 1;
            }

            if (strPFEmployee != "" && strPFCompany != "" && fun.NumberValidation(((TextBox)row.FindControl("txtPFEmployee0")).Text) == true && fun.NumberValidation(((TextBox)row.FindControl("txtPFCompany0")).Text) == true)
            {

                string strInsert = fun.update("tblHR_PF_Slab", "PFEmployee='" + strPFEmployee + "',PFCompany='" + strPFCompany + "',Active='" + strActive + "'", "Id='" + id + "'");

                SqlCommand cmd1 = new SqlCommand(strInsert, con);
                con.Open();
                cmd1.ExecuteNonQuery();
                con.Close();
                GridView1.EditIndex = -1;
                this.FillAccegrid();
            }
        }
        catch (Exception ex){}
    }
   

    protected void GridView1_RowEditing1(object sender, GridViewEditEventArgs e)
    {
        try
        {
            GridView1.EditIndex = e.NewEditIndex;
            this.FillAccegrid();
            int index = GridView1.EditIndex;
            GridViewRow grv = GridView1.Rows[index];
            int id = Convert.ToInt32(((Label)grv.FindControl("lblIDE")).Text);
            string sqlGetActive = fun.select("*", "tblHR_PF_Slab", "Id='" + id + "'");
            SqlCommand cmdGetActive = new SqlCommand(sqlGetActive, con);
            SqlDataAdapter DA = new SqlDataAdapter(cmdGetActive);
            DataSet DS = new DataSet();
            DA.Fill(DS);
            if (Convert.ToInt32(DS.Tables[0].Rows[0]["Active"]) == 1)
            {
                ((CheckBox)grv.FindControl("ChkActive0")).Checked = true;
                ((CheckBox)grv.FindControl("ChkActive0")).Enabled = true;
            }
            else
            {
                ((CheckBox)grv.FindControl("ChkActive0")).Checked = false;
                string sql2 = "SELECT Active FROM [tblHR_PF_Slab] where Active=1 order by [Id] desc";
                SqlCommand cmd2 = new SqlCommand(sql2, con);
                SqlDataAdapter da2 = new SqlDataAdapter(cmd2);
                DataSet ds2 = new DataSet();
                da2.Fill(ds2);
                if (ds2.Tables[0].Rows.Count > 0)
                {
                    ((CheckBox)grv.FindControl("ChkActive0")).Enabled = false;
                }
                else
                {
                    ((CheckBox)grv.FindControl("ChkActive0")).Enabled = true;
                }
            }
        }
        catch (Exception ex) { }


    }
    protected void GridView1_RowCancelingEdit(object sender, GridViewCancelEditEventArgs e)
    {
        GridView1.EditIndex =-1;
        this.FillAccegrid();
    }
    protected void GridView1_RowDeleting(object sender, GridViewDeleteEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);

            int id = Convert.ToInt32(GridView1.DataKeys[e.RowIndex].Value);
            SqlCommand cmd = new SqlCommand(fun.delete("tblHR_PF_Slab", "Id='" + id + "'"), con);
            con.Open();
            cmd.ExecuteNonQuery();
            con.Close();
            this.FillAccegrid();
           
        }
        catch (Exception er) { }
    }
}
