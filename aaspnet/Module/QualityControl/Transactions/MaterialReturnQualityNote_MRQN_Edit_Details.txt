    public void LoadData()
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        con.Open();
        
        try
        {
             DataTable dt = new DataTable();

            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Description", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("AcceptedQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("RetQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ItemId", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("TotalAccptedQty", typeof(double)));

            SqlCommand Cmdgrid = new SqlCommand(fun.select("tblQc_MaterialReturnQuality_Details.Id,tblQc_MaterialReturnQuality_Details.MRNId,tblQc_MaterialReturnQuality_Details.MRQNNo,tblQc_MaterialReturnQuality_Details.AcceptedQty,tblQc_MaterialReturnQuality_Master.MRNId as MSId", "tblQc_MaterialReturnQuality_Details,tblQc_MaterialReturnQuality_Master", "tblQc_MaterialReturnQuality_Master.Id='" + MId + "' AND tblQc_MaterialReturnQuality_Details.MId=tblQc_MaterialReturnQuality_Master.Id  AND tblQc_MaterialReturnQuality_Master.CompId='" + CompId + "'"), con);

            SqlDataReader DS = Cmdgrid.ExecuteReader();                       
            DataRow dr;

            while (DS.Read())
            {
                dr = dt.NewRow();

                {
                    dr[0] = Convert.ToInt32(DS["Id"]);

                    SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialReturn_Details.ItemId,tblInv_MaterialReturn_Details.DeptId,tblInv_MaterialReturn_Details.WONo,tblInv_MaterialReturn_Details.RetQty,tblInv_MaterialReturn_Details.Remarks", "tblInv_MaterialReturn_Details,tblInv_MaterialReturn_Master", "tblInv_MaterialReturn_Details.Id='" + Convert.ToInt32(DS["MRNId"]) + "' AND tblInv_MaterialReturn_Master.CompId='" + CompId + "'AND tblInv_MaterialReturn_Master.Id=tblInv_MaterialReturn_Details.MId"), con);

                    SqlDataReader dsrs = cmdrs.ExecuteReader();
                    dsrs.Read();

                    if (dsrs.HasRows==true)
                    {
                        SqlCommand cmditem = new SqlCommand(fun.select("ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs["ItemId"]) + "' AND  CompId='" + CompId + "' "), con);

                        SqlDataReader dsitem = cmditem.ExecuteReader();
                        dsitem.Read();

                        if (dsitem.HasRows == true)
                        {
                            SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOM", "Unit_Master", "Id='" + Convert.ToInt32(dsitem["UOMBasic"]) + "'"), con);
                            SqlDataReader dsunit = cmdunit.ExecuteReader();
                            dsunit.Read();

                            dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs["ItemId"].ToString()));
                            dr[2] = dsitem["ManfDesc"].ToString();

                            if (dsunit.HasRows == true)
                            {
                                dr[3] = dsunit["UOM"].ToString();
                            }
                            if (dsrs["DeptId"].ToString() == "0")
                            {
                                dr[4] = "NA";
                                dr[5] = dsrs["WONo"].ToString();
                            }
                            else
                            {

                                string dept = fun.select("Symbol", "tblHR_Departments", "Id='" + Convert.ToInt32(dsrs["DeptId"]) + "'");
                                SqlCommand cmddpt = new SqlCommand(dept, con);
                                SqlDataReader ds = cmddpt.ExecuteReader();
                                ds.Read();
                                
                                if (ds.HasRows == true)
                                {
                                    dr[4] = ds["Symbol"].ToString();
                                    dr[5] = "NA";
                                }
                            }

                            dr[7] = Convert.ToDouble(decimal.Parse((dsrs["RetQty"]).ToString()).ToString("N3"));
                            dr[6] = Convert.ToDouble(decimal.Parse((DS["AcceptedQty"]).ToString()).ToString("N3"));
                            dr[8] = dsrs["Remarks"].ToString();
                            dr[9] = Convert.ToInt32(dsrs["ItemId"]).ToString();

                            string sqlmindetails = fun.select("sum(tblQc_MaterialReturnQuality_Details.AcceptedQty) as sum_AcceptedQty", "tblQc_MaterialReturnQuality_Master,tblQc_MaterialReturnQuality_Details", "tblQc_MaterialReturnQuality_Master.CompId='" + CompId + "' AND tblQc_MaterialReturnQuality_Master.Id=tblQc_MaterialReturnQuality_Details.MId AND tblQc_MaterialReturnQuality_Details.MRNId='" + DS["MRNId"].ToString() + "' AND tblQc_MaterialReturnQuality_Master.MRNId='" + DS["MSId"].ToString() + "'");

                            SqlCommand cmdmindetails = new SqlCommand(sqlmindetails, con);
                            SqlDataReader DSmindetails = cmdmindetails.ExecuteReader();
                            DSmindetails.Read();

                            if (DSmindetails["sum_AcceptedQty"] != DBNull.Value)
                            {
                                dr[10] =Convert.ToDouble(decimal.Parse((DSmindetails["sum_AcceptedQty"]).ToString()).ToString("N3"));
                            }

                            else
                            {
                                dr[10] = "0";
                            }
                            dt.Rows.Add(dr);
                            dt.AcceptChanges();
                        }
                    }
                }
            }
            GridView1.DataSource = dt;
            GridView1.DataBind();
        }
        catch (Exception ex) { }
        finally
        { con.Close(); }
    }

    protected void GridView1_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);

            string sId = Session["username"].ToString();
            int CompId = Convert.ToInt32(Session["compid"]);
            int index = GridView1.EditIndex;
            string CDate = fun.getCurrDate();
            string CTime = fun.getCurrTime();

            GridViewRow row = GridView1.Rows[index];
            int id = Convert.ToInt32(((Label)row.FindControl("lblId")).Text);
            int ItemId = Convert.ToInt32(((Label)row.FindControl("lblItemId")).Text);
            string Remark = ((TextBox)row.FindControl("txtRemark")).Text;
            double AccptQty = Convert.ToDouble(decimal.Parse((((Label)row.FindControl("lblAccptQty1")).Text).ToString()).ToString("N3"));
            double BalStkQty = 0;
            double Qty = Convert.ToDouble(decimal.Parse((((TextBox)row.FindControl("txtAccQty")).Text).ToString()).ToString("N3"));

            con.Open();

            if (AccptQty >= Qty && fun.NumberValidationQty(Qty.ToString()) == true)
            {
                double DiffQty = AccptQty - Qty;

                string sqlstkqty = fun.select("StockQty", "tblDG_Item_Master", "CompId='" + CompId + "' AND Id='" + ItemId + "'");
                SqlCommand cmd5 = new SqlCommand(sqlstkqty, con);
                SqlDataAdapter dastk5 = new SqlDataAdapter(cmd5);
                DataSet dsstk5 = new DataSet();
                dastk5.Fill(dsstk5);

                if (dsstk5.Tables[0].Rows.Count > 0)
                {
                    double StQty = Convert.ToDouble(decimal.Parse((dsstk5.Tables[0].Rows[0][0]).ToString()).ToString("N3"));

                    if (StQty >= DiffQty )
                    {
                        string sql = fun.update("tblQc_MaterialReturnQuality_Details", "AcceptedQty='" + Qty + "'", "Id='" + id + "'");
                        SqlCommand cmd = new SqlCommand(sql, con);
                        cmd.ExecuteNonQuery();

                        string sql2 = fun.update("tblQc_MaterialReturnQuality_Master", "SysDate='" + CDate + "',SysTime='" + CTime + "',SessionId='" + sId + "'", "Id='" + MId + "' And CompId='" + CompId + "'");
                        SqlCommand cmd2 = new SqlCommand(sql2, con);
                        cmd2.ExecuteNonQuery();
                        BalStkQty = StQty - DiffQty;

                        string update = fun.update("tblDG_Item_Master", "StockQty='" + BalStkQty + "'", "CompId='" + CompId + "' AND Id='" + ItemId + "'");
                        SqlCommand cmd4 = new SqlCommand(update, con);
                        cmd4.ExecuteNonQuery();

                    }
                }
                con.Close();
                GridView1.EditIndex = -1;
                this.LoadData();
            }

        }
        catch (Exception ex)
        {

        }
    }
