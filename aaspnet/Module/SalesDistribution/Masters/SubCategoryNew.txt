﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_SalesDistribution_Masters_SubCategoryNew : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    protected void Page_Load(object sender, EventArgs e)
    {
        lblMessage.Text = "";

    }
    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);

            if (e.CommandName == "Add")
            {
                int strCId = Convert.ToInt32(((DropDownList)GridView1.FooterRow.FindControl("ddCategory")).SelectedValue);
                string strScName = ((TextBox)
                GridView1.FooterRow.FindControl("txtSCName")).Text;
                string strsymbol = ((TextBox)GridView1.FooterRow.FindControl("txtSymbol")).Text;
                string CDate = fun.getCurrDate();
                string CTime = fun.getCurrTime();
                string sId = Session["username"].ToString();
                int CompId = Convert.ToInt32(Session["compid"]);
                int FinYearId = Convert.ToInt32(Session["finyear"]);
                string sql = fun.select("Symbol", "tblSD_WO_SubCategory", "CompId='" + CompId + "' And CId='" + strCId + "' AND  Symbol='" + strsymbol + "'And FinYearId<='" + FinYearId + "' Order by CId desc");
                SqlCommand cmd1 = new SqlCommand(sql, con);
                SqlDataAdapter da = new SqlDataAdapter(cmd1);
                DataSet ds = new DataSet();
                da.Fill(ds);
                if (ds.Tables[0].Rows.Count == 0)
                {
                    if (strScName != "" && strsymbol != "")
                    {
                        LocalSqlServer.InsertParameters["SysDate"].DefaultValue = CDate.ToString();
                        LocalSqlServer.InsertParameters["SysTime"].DefaultValue = CTime;
                        LocalSqlServer.InsertParameters["CompId"].DefaultValue = CompId.ToString();
                        LocalSqlServer.InsertParameters["FinYearId"].DefaultValue = FinYearId.ToString();
                        LocalSqlServer.InsertParameters["SessionId"].DefaultValue = sId;
                        LocalSqlServer.InsertParameters["CId"].DefaultValue = strCId.ToString();
                        LocalSqlServer.InsertParameters["SCName"].DefaultValue = strScName;
                        LocalSqlServer.InsertParameters["Symbol"].DefaultValue = strsymbol.ToUpper();
                        LocalSqlServer.Insert();
                        lblMessage.Text = "Record Inserted.";
                    }
                }
                else
                {

                    string mystring = string.Empty;
                    mystring = "SubCategory symbol is already used.";
                    ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
                }


            }
            else if (e.CommandName == "Add1")
            {

                int strCId = Convert.ToInt32(((DropDownList)GridView1.Controls[0].Controls[0].FindControl("ddCategory")).SelectedValue);
                string strScName = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtSCName")).Text;
                string strsymbol = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtSymbol")).Text;
                string CDate = fun.getCurrDate();
                string CTime = fun.getCurrTime();
                string sId = Session["username"].ToString();
                int CompId = Convert.ToInt32(Session["compid"]);
                int FinYearId = Convert.ToInt32(Session["finyear"]);
                if (strScName != "" && strsymbol != "")
                {
                    LocalSqlServer.InsertParameters["SysDate"].DefaultValue = CDate.ToString();
                    LocalSqlServer.InsertParameters["SysTime"].DefaultValue = CTime;
                    LocalSqlServer.InsertParameters["CompId"].DefaultValue = CompId.ToString();
                    LocalSqlServer.InsertParameters["FinYearId"].DefaultValue = FinYearId.ToString();
                    LocalSqlServer.InsertParameters["SessionId"].DefaultValue = sId;
                    LocalSqlServer.InsertParameters["CId"].DefaultValue = strCId.ToString();
                    LocalSqlServer.InsertParameters["SCName"].DefaultValue = strScName;
                    LocalSqlServer.InsertParameters["Symbol"].DefaultValue = strsymbol.ToUpper();
                    LocalSqlServer.Insert();
                    lblMessage.Text = "Record Inserted.";
                }
            }
        }
        catch (Exception ex)
        {
        }



    }
}
