﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_HR_Transactions_NewsandNotices_Delete : System.Web.UI.Page
{

    clsFunctions fun = new clsFunctions();
    protected void Page_Load(object sender, EventArgs e)
    {

    }
   
    protected void GridView2_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        int CompId = Convert.ToInt32(Session["compid"]);
        int FinYearId = Convert.ToInt32(Session["finyear"]);

        if (e.CommandName == "Del")
        {
            con.Open();
            try
            {               
                foreach (GridViewRow grv in GridView2.Rows)
                {
                    int id = Convert.ToInt32(((Label)grv.FindControl("lblId")).Text);
                    string delete = fun.delete("tblHR_News_Notices", "CompId='" + CompId + "' AND FinYearId<='" + FinYearId + "' AND Id='" +id + "'");
                    SqlCommand cmd = new SqlCommand(delete, con);
                    cmd.ExecuteNonQuery();
                    Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }               
            }
            catch (Exception ex) { }
            finally
            {
                con.Close();

            }
        }
    }


    protected void GridView2_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        if (e.Row.RowType == DataControlRowType.DataRow)
        {
            LinkButton del = (LinkButton)e.Row.FindControl("LinkButton1");
            del.Attributes.Add("onclick", "return confirmationDelete();");
        }

    }
}
