    protected void Delete_Click(object sender, EventArgs e)

        SqlConnection con = new SqlConnection(connStr);       
        try
        {
            con.Open();           
            string cmdstr = fun.delete("SD_Cust_Enquiry_Master", "CustomerId='" + CustCode + "' and EnqId='" + EnqId + "' and CompId='"+CId+"'");
            SqlCommand cmd = new SqlCommand(cmdstr, con);
            cmd.ExecuteNonQuery(); 
            Response.Redirect("CustEnquiry_Delete.aspx?ModId=2&SubModId=10");
            
        }

        catch (SqlException se)
        {
            string mystring = string.Empty;
            mystring = "You Cannot delete this Enquiry. This is in use. ";
            ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
        }


        catch (Exception ex)
        { }
        finally
        {
            con.Close();
           
        }
    }
