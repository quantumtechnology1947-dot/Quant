﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_Inventory_Transactions_MaterialRequisitionSlip_Dashboard : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    string sId = "";
    int CompId = 0;

    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            sId = Session["username"].ToString();
            CompId = Convert.ToInt32(Session["compid"]);
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            con.Open();
            string delTemp = fun.delete("tblinv_MaterialRequisition_Temp", "CompId='" + CompId + "'And SessionId='" + sId + "'");
            SqlCommand cmdTemp = new SqlCommand(delTemp, con);
            cmdTemp.ExecuteNonQuery();
            con.Close();
        }
        catch (Exception es)
        {

        }
    }
}
