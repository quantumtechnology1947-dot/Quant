﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;
using CrystalDecisions.CrystalReports.Engine;

public partial class Module_QualityControl_Transactions_MaterialReturnQualityNote_MRQN_Print_Details : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    ReportDocument cryRpt = new ReportDocument();
    DataSet DS = new DataSet();
    string MRNNo = "";
    string MRQNNo = "";
    int FyId = 0;
    int CompId = 0;    
    string MId = "";
    string Key = string.Empty;

    protected void Page_Init(object sender, EventArgs e)
    {
        if (!IsPostBack)
        {

            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            con.Open();

            DataTable dt = new DataTable();
            DataSet Mrqn = new DataSet();

            try
            {

                MRNNo = Request.QueryString["MRNNo"].ToString();
                MRQNNo = Request.QueryString["MRQNNo"].ToString();
                FyId = Convert.ToInt32(Request.QueryString["FYId"]);
                CompId = Convert.ToInt32(Session["compid"]);
                MId = Request.QueryString["Id"].ToString();
                Key = Request.QueryString["Key"].ToString();


                SqlCommand Cmdgrid = new SqlCommand(fun.select("tblQc_MaterialReturnQuality_Details.Id,tblQc_MaterialReturnQuality_Details.MRNId,tblQc_MaterialReturnQuality_Details.MRQNNo,tblQc_MaterialReturnQuality_Details.AcceptedQty,tblQc_MaterialReturnQuality_Master.MRNId as MSId", "tblQc_MaterialReturnQuality_Details,tblQc_MaterialReturnQuality_Master", "tblQc_MaterialReturnQuality_Master.Id='" + MId + "' AND tblQc_MaterialReturnQuality_Details.MId=tblQc_MaterialReturnQuality_Master.Id  AND tblQc_MaterialReturnQuality_Master.CompId='" + CompId + "'"), con);

                SqlDataReader DS = Cmdgrid.ExecuteReader();

                dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
                dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("Description", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("AcceptedQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("RetQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("CompId", typeof(int)));

          
                DataRow dr;
                while (DS.Read())
                {
                    dr = dt.NewRow();

                    dr[0] = Convert.ToInt32(DS["Id"].ToString());

                    SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialReturn_Details.ItemId,tblInv_MaterialReturn_Details.DeptId,tblInv_MaterialReturn_Details.WONo,tblInv_MaterialReturn_Details.RetQty,tblInv_MaterialReturn_Details.Remarks", "tblInv_MaterialReturn_Details,tblInv_MaterialReturn_Master", "tblInv_MaterialReturn_Details.Id='" + Convert.ToInt32(DS["MRNId"]) + "' AND tblInv_MaterialReturn_Master.CompId='" + CompId + "'AND tblInv_MaterialReturn_Master.Id=tblInv_MaterialReturn_Details.MId"), con);
                    SqlDataReader dsrs = cmdrs.ExecuteReader();
                    dsrs.Read();

                    if (dsrs.HasRows == true)
                    {
                        SqlCommand cmditem = new SqlCommand(fun.select("Id,ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs["ItemId"]) + "' AND  CompId='" + CompId + "' "), con);
                        SqlDataReader dsitem = cmditem.ExecuteReader();
                        dsitem.Read();

                        if (dsitem.HasRows == true)
                        {
                            SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOM", "Unit_Master", "Id='" + Convert.ToInt32(dsitem["UOMBasic"]) + "'"), con);
                            SqlDataReader dsunit = cmdunit.ExecuteReader();
                            dsunit.Read();

                            dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs["ItemId"].ToString()));
                            dr[2] = dsitem["ManfDesc"].ToString();
                            if (dsunit.HasRows == true)
                            {
                                dr[3] = dsunit["UOM"].ToString();
                            }
                            if (dsrs["DeptId"].ToString() == "0")
                            {
                                dr[4] = "NA";
                                dr[5] = dsrs["WONo"].ToString();
                            }
                            else
                            {
                                string dept = fun.select("Symbol", "tblHR_Departments", "Id='" + Convert.ToInt32(dsrs["DeptId"]) + "'");
                                SqlCommand cmddpt = new SqlCommand(dept, con);
                                SqlDataReader ds = cmddpt.ExecuteReader();
                                ds.Read();

                                if (ds.HasRows == true)
                                {
                                    dr[4] = ds["Symbol"].ToString();
                                    dr[5] = "NA";
                                }
                            }
                            dr[6] = Convert.ToDouble(decimal.Parse((dsrs["RetQty"]).ToString()).ToString("N3"));
                            dr[7] = Convert.ToDouble(decimal.Parse((DS["AcceptedQty"]).ToString()).ToString("N3"));
                            dr[8] = dsrs["Remarks"].ToString();
                            dr[9] = CompId;
                            dt.Rows.Add(dr);
                            dt.AcceptChanges();
                        }
                    }
                }

                Mrqn.Tables.Add(dt);
                DataSet xsdds = new MRS();
                xsdds.Tables[0].Merge(Mrqn.Tables[0]);
                xsdds.AcceptChanges();
                cryRpt = new ReportDocument();
                cryRpt.Load(Server.MapPath("~/Module/QualityControl/Transactions/Reports/MRQN_Print.rpt"));
                cryRpt.SetDataSource(xsdds);
                // For MRQN Date
                string sqlMRQNDate = fun.select("SessionId,SysDate", "tblQc_MaterialReturnQuality_Master", "Id='" + MId + "' AND CompId='" + CompId + "' ");
                SqlCommand cmdMRQNDate = new SqlCommand(sqlMRQNDate, con);
                SqlDataReader DSMRQNDate = cmdMRQNDate.ExecuteReader();
                DSMRQNDate.Read();

                if (DSMRQNDate.HasRows == true)
                {
                    string MRQNDate = fun.FromDateDMY(DSMRQNDate["SysDate"].ToString());
                    cryRpt.SetParameterValue("MRQNDate", MRQNDate);
                }
                // For Gen By.
                string sqlGenBy = fun.select("Title+'. '+EmployeeName As EmpName", "tblHR_OfficeStaff", "CompId='" + CompId + "' AND EmpId='" + DSMRQNDate["SessionId"] + "'");
                SqlCommand cmdGenBy = new SqlCommand(sqlGenBy, con);
                SqlDataReader DSGenBy = cmdGenBy.ExecuteReader();
                DSGenBy.Read();
                if (DSGenBy.HasRows == true)
                {
                    string GenBy = DSGenBy["EmpName"].ToString();
                    cryRpt.SetParameterValue("GenBy", GenBy);
                }

                // For Address
                string Address = fun.CompAdd(CompId);
                cryRpt.SetParameterValue("Address", Address);
                // For MRNNo And MRQNNo
                cryRpt.SetParameterValue("MRNNo", MRNNo);
                cryRpt.SetParameterValue("MRQNNo", MRQNNo);
                CrystalReportViewer1.ReportSource = cryRpt;
                Session[Key] = cryRpt;
            }
            catch (Exception ex) { }
            finally
            {
                Mrqn.Clear();
                Mrqn.Dispose();
                dt.Clear();
                dt.Dispose();
                con.Close();
                con.Dispose();

            }
        }

        else
        {
            Key = Request.QueryString["Key"].ToString();
            ReportDocument doc = (ReportDocument)Session[Key];
            CrystalReportViewer1.ReportSource = doc;
        }
    }
    protected void Page_Load(object sender, EventArgs e)
    {
        cryRpt = new ReportDocument();

    }
    protected void btnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("MaterialReturnQualityNote_MRQN_Print.aspx?ModId=10&SubModId=49");
    }
    protected void Page_UnLoad(object sender, EventArgs e)
    {
        this.CrystalReportViewer1.Dispose();
        this.CrystalReportViewer1 = null;
        cryRpt.Close();
        cryRpt.Dispose();
        GC.Collect();
    }

}
