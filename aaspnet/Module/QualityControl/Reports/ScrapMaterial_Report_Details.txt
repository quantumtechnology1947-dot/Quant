﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;
using CrystalDecisions.CrystalReports.Engine;

public partial class Module_QualityControl_Reports_ScrapMaterial_Report_Details : System.Web.UI.Page
{

    clsFunctions fun = new clsFunctions();
   
    string MRNNo = "";
    string MRQNNo = "";
    int FyId = 0;
    int CompId = 0;   
    string ScrNo = "";
    string sId = "";
    int FinYearId = 0;
    ReportDocument cryRpt = new ReportDocument();
    string Key = string.Empty;

    protected void Page_Init(object sender, EventArgs e)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        DataSet SCR = new DataSet();
        DataTable dt = new DataTable();
        con.Open();
        if (!IsPostBack)
        {
            try
            {
                sId = Session["username"].ToString();
                FinYearId = Convert.ToInt32(Session["finyear"]);
                CompId = Convert.ToInt32(Session["compid"]);
                MRNNo = Request.QueryString["MRNNo"].ToString();
                MRQNNo = Request.QueryString["MRQNNo"].ToString();
                FyId = Convert.ToInt32(Request.QueryString["FYId"]);
                CompId = Convert.ToInt32(Session["compid"]);
                ScrNo = Request.QueryString["ScrapNo"].ToString();
                Key = Request.QueryString["Key"].ToString();

                string StrSC = fun.select("*", "tblQC_Scrapregister", "FinYearId<='" + FinYearId + "' And CompId='" + CompId + "' And ScrapNo='" + ScrNo + "' ");
                SqlCommand sqlsc = new SqlCommand(StrSC, con);
                SqlDataReader dssc = sqlsc.ExecuteReader();
             
                dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
                dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("ManfDesc", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("UOMBasic", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("AcceptedQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("RetQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("Qty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("ScrapNo", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("CompId", typeof(int)));
                dt.Columns.Add(new System.Data.DataColumn("SysDate", typeof(string)));

                DataRow dr;
                while (dssc.Read())
                {
                    dr = dt.NewRow();

                    {
                        dr[0] = Convert.ToInt32(dssc["Id"]);

                        SqlCommand Cmdgrid = new SqlCommand(fun.select("tblQc_MaterialReturnQuality_Details.Id,tblQc_MaterialReturnQuality_Details.MRNId,tblQc_MaterialReturnQuality_Details.MRQNNo,tblQc_MaterialReturnQuality_Details.AcceptedQty,tblQc_MaterialReturnQuality_Master.MRNId as MSId", "tblQc_MaterialReturnQuality_Details,tblQc_MaterialReturnQuality_Master", "tblQc_MaterialReturnQuality_Master.Id='" + dssc["MRQNId"].ToString() + "' AND tblQc_MaterialReturnQuality_Details.MId=tblQc_MaterialReturnQuality_Master.Id  AND tblQc_MaterialReturnQuality_Master.CompId='" + CompId + "'"), con);
                        SqlDataReader DS = Cmdgrid.ExecuteReader();
                        DS.Read();

                        if (DS.HasRows == true)
                        {
                            SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialReturn_Details.ItemId,tblInv_MaterialReturn_Details.DeptId,tblInv_MaterialReturn_Details.WONo,tblInv_MaterialReturn_Details.RetQty,tblInv_MaterialReturn_Details.Remarks", "tblInv_MaterialReturn_Details,tblInv_MaterialReturn_Master", "tblInv_MaterialReturn_Details.Id='" + Convert.ToInt32(DS["MRNId"]) + "' AND tblInv_MaterialReturn_Master.CompId='" + CompId + "'AND tblInv_MaterialReturn_Master.Id=tblInv_MaterialReturn_Details.MId"), con);

                            SqlDataReader dsrs = cmdrs.ExecuteReader();
                            dsrs.Read();

                            if (dsrs.HasRows == true)
                            {
                                SqlCommand cmditem = new SqlCommand(fun.select("ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs["ItemId"]) + "' AND  CompId='" + CompId + "' "), con);
                                SqlDataReader dsitem = cmditem.ExecuteReader();
                                dsitem.Read();

                                if (dsitem.HasRows==true)
                                {
                                    SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOMBasic", "Unit_Master", "Id='" + Convert.ToInt32(dsitem["UOMBasic"]) + "'"), con);
                                    SqlDataReader dsunit = cmdunit.ExecuteReader();
                                    dsunit.Read();

                                    dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs["ItemId"].ToString()));
                                    dr[2] = dsitem["ManfDesc"].ToString();

                                    if (dsunit.HasRows==true)
                                    {
                                        dr[3] = dsunit["UOMBasic"].ToString();
                                    }
                                    if (dsrs["DeptId"].ToString() == "0")
                                    {
                                        dr[4] = "NA";
                                        dr[5] = dsrs["WONo"].ToString();
                                    }
                                    else
                                    {
                                        string dept = fun.select("Symbol", "tblHR_Departments", "Id='" + Convert.ToInt32(dsrs["DeptId"]) + "'");
                                        SqlCommand cmddpt = new SqlCommand(dept, con);
                                        SqlDataReader ds = cmddpt.ExecuteReader();
                                        ds.Read();
                                       
                                        if (ds.HasRows==true)
                                        {
                                            dr[4] = ds["Symbol"].ToString();
                                            dr[5] = "NA";
                                        }
                                    }
                                    string sqlmindetails = fun.select("sum(tblQc_MaterialReturnQuality_Details.AcceptedQty) as AcceptedQty", "tblQc_MaterialReturnQuality_Master,tblQc_MaterialReturnQuality_Details", "tblQc_MaterialReturnQuality_Master.CompId='" + CompId + "' AND tblQc_MaterialReturnQuality_Master.Id=tblQc_MaterialReturnQuality_Details.MId AND tblQc_MaterialReturnQuality_Details.MRNId='" + DS["MRNId"].ToString() + "' AND tblQc_MaterialReturnQuality_Master.MRNId='" + DS["MSId"].ToString() + "'");

                                    SqlCommand cmdmindetails = new SqlCommand(sqlmindetails, con);
                                    SqlDataReader DSmindetails = cmdmindetails.ExecuteReader();
                                    DSmindetails.Read();
                                            
                                    if (DSmindetails["AcceptedQty"] != DBNull.Value)
                                    {
                                        dr[6] = Convert.ToDouble(decimal.Parse((DSmindetails["AcceptedQty"]).ToString()).ToString("N3"));
                                    }
                                    else
                                    {
                                        dr[10] = "0";
                                    }

                                    dr[7] = Convert.ToDouble(decimal.Parse((dsrs["RetQty"]).ToString()).ToString("N3"));
                                    dr[8] = dsrs["Remarks"].ToString();
                                    dr[9] = Convert.ToDouble(decimal.Parse((dssc["Qty"]).ToString()).ToString("N3"));
                                    dr[10] = (dssc["ScrapNo"].ToString());
                                    dr[11] = Convert.ToInt32(dssc["CompId"]);
                                    dr[12] = fun.FromDateDMY(dssc["SysDate"].ToString());

                                    dt.Rows.Add(dr);
                                    dt.AcceptChanges();
                                }
                            }
                        }
                    }
                }



                SCR.Tables.Add(dt);
                DataSet xsdds = new Scrap();
                xsdds.Tables[0].Merge(SCR.Tables[0]);

                xsdds.AcceptChanges();
                cryRpt = new ReportDocument();
                cryRpt.Load(Server.MapPath("~/Module/QualityControl/Reports/ScrapReport.rpt"));
                cryRpt.SetDataSource(xsdds);
                CrystalReportSource1.DataBind();
                // For Address
                string Address = fun.CompAdd(CompId);
                cryRpt.SetParameterValue("Address", Address);
                CrystalReportViewer1.ReportSource = cryRpt;
                Session[Key] = cryRpt;

            }
            catch (Exception ex) { }
            finally
            {
                SCR.Clear();
                SCR.Dispose();
                dt.Clear();
                dt.Dispose();
                con.Close();
                con.Dispose();

            }

        }

        else
        {
            Key = Request.QueryString["Key"].ToString();
            ReportDocument doc = (ReportDocument)Session[Key];
            CrystalReportViewer1.ReportSource = doc;
        }
    }
    protected void Page_Load(object sender, EventArgs e)
    {
        cryRpt = new ReportDocument();
    } 
    protected void btnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("ScrapMaterial_Report.aspx?ModId=10&SubModId=");
    }
    protected void Page_UnLoad(object sender, EventArgs e)
    {
        this.CrystalReportViewer1.Dispose();
        this.CrystalReportViewer1 = null;
        cryRpt.Close();
        cryRpt.Dispose();
        GC.Collect();
    }

}
