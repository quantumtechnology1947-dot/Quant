    }
    protected void GridView1_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            int index = GridView1.EditIndex;
            GridViewRow row = GridView1.Rows[index];
            string StrDesc = (((TextBox)row.FindControl("txtDescription1")).Text);
            string StrItemCode = (((TextBox)row.FindControl("txtItemCode1")).Text);
            string StrQty = (((TextBox)row.FindControl("txtQty1")).Text);
            if (StrDesc != "" && StrItemCode != "" && StrQty != "")
            {
                SqlDataSource1.UpdateParameters["Description"].DefaultValue = StrDesc;
                SqlDataSource1.UpdateParameters["ItemCode"].DefaultValue = StrItemCode;
                SqlDataSource1.UpdateParameters["Qty"].DefaultValue = StrQty;
                SqlDataSource1.Update();

            }
        }
        catch (Exception ex) { }
    }
