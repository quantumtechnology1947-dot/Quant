﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_HR_Masters_HolidayMaster : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    int CompId = 0;
    int FinYearId = 0;
    string CDate = "";
    string CTime = "";
    string sId = "";
    protected void Page_Load(object sender, EventArgs e)
    {
        CompId = Convert.ToInt32(Session["compid"]);
        FinYearId = Convert.ToInt32(Session["finyear"]);
        sId = Session["username"].ToString();
        CDate = fun.getCurrDate();
        CTime = fun.getCurrTime();
        lblMessage.Text = "";      
          
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {

        try
        {
            if (e.CommandName == "Add")
            {
                string strTitle0 = ((TextBox)GridView1.FooterRow.FindControl("txtTitle0")).Text;
                string strDate0 = fun.FromDate(((TextBox)GridView1.FooterRow.FindControl("TxtDate0")).Text);
                if (strTitle0 != "" && strDate0 != "")
                {
                    SqlDataSource1.InsertParameters["Title"].DefaultValue = strTitle0;
                    SqlDataSource1.InsertParameters["HDate"].DefaultValue = strDate0;
                    SqlDataSource1.InsertParameters["SysDate"].DefaultValue = CDate;
                    SqlDataSource1.InsertParameters["SysTime"].DefaultValue = CTime;
                    SqlDataSource1.Insert();
                    lblMessage.Text = "Record Inserted";
                }
            }
            else if (e.CommandName == "Add1")
            {
                string strTitle1 = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtTitle1")).Text;
                string strDate1 = fun.FromDate(((TextBox)GridView1.Controls[0].Controls[0].FindControl("TxtDate1")).Text);
                if (strTitle1 != "" && strDate1 != "")
                {
                    SqlDataSource1.InsertParameters["Title"].DefaultValue = strTitle1;
                    SqlDataSource1.InsertParameters["HDate"].DefaultValue = strDate1;
                    SqlDataSource1.InsertParameters["SysDate"].DefaultValue = CDate;
                    SqlDataSource1.InsertParameters["SysTime"].DefaultValue = CTime;
                    SqlDataSource1.Insert();
                    lblMessage.Text = "Record Inserted";
                }

            }
        }

        catch (Exception ex) { }

    }
    protected void GridView1_RowDeleted(object sender, GridViewDeletedEventArgs e)
    {
        lblMessage.Text = "Record Deleted";
    }
    protected void GridView1_RowUpdated(object sender, GridViewUpdatedEventArgs e)
    {
        lblMessage.Text = "Record Updated";
    }

    protected void GridView1_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        try
        {
            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del = (LinkButton)e.Row.Cells[0].Controls[0];
                del.Attributes.Add("onclick", "return confirmationUpdate();");
            }

            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del1 = (LinkButton)e.Row.Cells[1].Controls[0];
                del1.Attributes.Add("onclick", "return confirmationDelete();");
            }


            if (GridView1.Rows.Count == 0)
            {
                ((TextBox)GridView1.Controls[0].Controls[0].FindControl("TxtDate1")).Attributes.Add("readonly", "readonly");
            }
            else
            {
                ((TextBox)GridView1.FooterRow.FindControl("TxtDate0")).Attributes.Add("readonly", "readonly");
           
                 GridViewRow row = GridView1.Rows[GridView1.EditIndex];              
                ((TextBox)row.FindControl("TxtDate")).Attributes.Add("readonly", "readonly");
            }

        }
        catch (Exception ex) { }

    }
    protected void GridView1_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            int rowIndex = GridView1.EditIndex;
            GridViewRow row = GridView1.Rows[rowIndex];
            

            string StrTitle = ((TextBox)row.FindControl("TxtTitle")).Text;
            string StrDate = fun.FromDate(((TextBox)row.FindControl("TxtDate")).Text);
            SqlDataSource1.UpdateParameters["SysDate"].DefaultValue = CDate;
            SqlDataSource1.UpdateParameters["SysTime"].DefaultValue = CTime;
            SqlDataSource1.UpdateParameters["Title"].DefaultValue = StrTitle;
            SqlDataSource1.UpdateParameters["HDate"].DefaultValue = StrDate;
            SqlDataSource1.Update();
            lblMessage.Text = "Record Inserted";

        }

        catch (Exception ex) { }
    }
}
