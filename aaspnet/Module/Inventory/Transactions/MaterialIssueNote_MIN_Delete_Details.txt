﻿using System;
using System.Collections;
using System.Configuration;

using System.Linq;

public partial class Module_Inventory_Transactions_MaterialIssueNote_MIN_Delete_Details : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    DataSet DS = new DataSet();
    SqlConnection con;
    string MINNo = "";
    string MRSNo = "";
    int FyId = 0;
    int CompId = 0;
    string sId = "";
    string connStr = "";
    string MId = "";
    protected void Page_Load(object sender, EventArgs e)
    {
       try
        {
            MId = Request.QueryString["Id"].ToString();
            MINNo = Request.QueryString["MINNo"].ToString();
            MRSNo = Request.QueryString["MRSNo"].ToString();
            FyId = Convert.ToInt32(Request.QueryString["FYId"]);
            CompId = Convert.ToInt32(Session["compid"]);
            sId = Session["username"].ToString();
            connStr = fun.Connection();
            con = new SqlConnection(connStr);

            this.LoadData();
        }
        catch(Exception ex){}

    }

    public void LoadData()
    {
        try
        {
            con.Open();
            SqlCommand Cmdgrid = new SqlCommand(fun.select("tblInv_MaterialIssue_Details.Id,tblInv_MaterialIssue_Details.MRSId,tblInv_MaterialIssue_Master.MRSId as MMId,tblInv_MaterialIssue_Details.IssueQty", "tblInv_MaterialIssue_Details,tblInv_MaterialIssue_Master", "tblInv_MaterialIssue_Master.Id='" + MId + "' AND tblInv_MaterialIssue_Master.Id=tblInv_MaterialIssue_Details.MId AND tblInv_MaterialIssue_Master.CompId='" + CompId + "'"), con);

            SqlDataReader dagrid = Cmdgrid.ExecuteReader();            
            DataTable dt = new DataTable();
            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(Int32)));
            dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ManfDesc", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ReqQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("IssueQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ItemId", typeof(string)));
            DataRow dr;           
            while (dagrid.Read())
            {

                dr = dt.NewRow();

                dr[0] = Convert.ToInt32(dagrid["Id"]);

                SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialRequisition_Details.ItemId,tblInv_MaterialRequisition_Details.DeptId,tblInv_MaterialRequisition_Details.WONo,tblInv_MaterialRequisition_Details.ReqQty,tblInv_MaterialRequisition_Details.Remarks", "tblInv_MaterialRequisition_Master,tblInv_MaterialRequisition_Details", "tblInv_MaterialRequisition_Details.Id='" + Convert.ToInt32(dagrid["MRSId"]) + "' AND tblInv_MaterialRequisition_Master.CompId='" + CompId + "' AND tblInv_MaterialRequisition_Master.Id='" + dagrid["MMId"] + "' AND tblInv_MaterialRequisition_Master.Id=tblInv_MaterialRequisition_Details.MId"), con);

                SqlDataAdapter dars = new SqlDataAdapter(cmdrs);
                DataSet dsrs = new DataSet();
                dars.Fill(dsrs);

                if (dsrs.Tables[0].Rows.Count > 0)
                {
                    SqlCommand cmditem = new SqlCommand(fun.select("ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"]) + "' AND  CompId='" + CompId + "'"), con);
                    SqlDataAdapter daitem = new SqlDataAdapter(cmditem);
                    DataSet dsitem = new DataSet();
                    daitem.Fill(dsitem);
                    if (dsitem.Tables[0].Rows.Count > 0)
                    {
                        SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOM", "Unit_Master", "Id='" + Convert.ToInt32(dsitem.Tables[0].Rows[0]["UOMBasic"]) + "'"), con);
                        DataSet dsunit = new DataSet();
                        SqlDataAdapter daunit = new SqlDataAdapter(cmdunit);
                        daunit.Fill(dsunit);

                        if (dsunit.Tables[0].Rows.Count > 0)
                        {
                            dr[3] = dsunit.Tables[0].Rows[0]["UOM"].ToString();
                        }

                        dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"].ToString()));
                        dr[2] = dsitem.Tables[0].Rows[0]["ManfDesc"].ToString();
                    }


                    if (dsrs.Tables[0].Rows[0]["DeptId"].ToString() == "1")
                    {
                        dr[4] = "NA";
                        dr[5] = dsrs.Tables[0].Rows[0]["WONo"].ToString();
                    }
                    else
                    {
                        if (dsrs.Tables[0].Rows[0]["DeptId"] != DBNull.Value)
                        {
                            string dept = fun.select("Symbol", "BusinessGroup", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["DeptId"]) + "'");

                            SqlCommand cmddpt = new SqlCommand(dept, con);
                            DataSet ds = new DataSet();
                            SqlDataAdapter dadpt = new SqlDataAdapter(cmddpt);
                            dadpt.Fill(ds);
                            if (ds.Tables[0].Rows.Count > 0)
                            {
                                dr[4] = ds.Tables[0].Rows[0]["Symbol"].ToString();
                                dr[5] = "NA";
                            }
                        }
                        else
                        {
                            dr[5] = dsrs.Tables[0].Rows[0]["WONo"].ToString();
                            dr[4] = "NA";
                        }
                    }

                    dr[6] = Convert.ToDouble(decimal.Parse(dsrs.Tables[0].Rows[0]["ReqQty"].ToString()).ToString("N3"));
                    dr[7] = Convert.ToDouble(decimal.Parse(dagrid["IssueQty"].ToString()).ToString("N3"));
                    dr[8] = dsrs.Tables[0].Rows[0]["Remarks"].ToString();
                    dr[9] = dsrs.Tables[0].Rows[0]["ItemId"].ToString();
                }
                dt.Rows.Add(dr);
                dt.AcceptChanges();
            }
            GridView1.DataSource = dt;
            GridView1.DataBind();
        }
        catch (Exception ex) { }
        finally {con.Close();}
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {
       
       try
        {
            con.Open();
            if (e.CommandName == "Del")
            {
               
                GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
                int Id = Convert.ToInt32(((Label)row.FindControl("lblId")).Text);
                string ItemId = ((Label)row.FindControl("lblItemId")).Text;
                double StkQty = 0;
                double IssuedQty = 0;
                double BalStkQty = 0;                                
                string StrStkQty = fun.select("StockQty", "tblDG_Item_Master", "Id='" + ItemId + "' AND CompId='" + CompId + "'");
                SqlCommand cmdStkQty = new SqlCommand(StrStkQty, con);
                SqlDataAdapter daStkQty = new SqlDataAdapter(cmdStkQty);
                DataSet DSStkQty = new DataSet();
                daStkQty.Fill(DSStkQty);
                if (DSStkQty.Tables[0].Rows.Count > 0)
                {
                    StkQty = Convert.ToDouble(decimal.Parse(DSStkQty.Tables[0].Rows[0][0].ToString()).ToString("N3"));
                }
                IssuedQty = Convert.ToDouble(decimal.Parse(((Label)row.FindControl("lblIssueQty")).Text).ToString("N3")); 
                BalStkQty = StkQty + IssuedQty;

                //Generate Log For Transaction Type: MIN - 2
                fun.StkAdjLog(CompId, FyId, sId, 2, MINNo, Convert.ToInt32(ItemId), IssuedQty);

                string delStr = fun.delete("tblInv_MaterialIssue_Details", "Id='" + Id + "' AND MId='"+MId+"'");
                SqlCommand cmdDel = new SqlCommand(delStr, con);
                cmdDel.ExecuteNonQuery();

                string StrUpStkQty = fun.update("tblDG_Item_Master", "StockQty='" + BalStkQty + "'", "Id='" + ItemId + "' AND CompId='" + CompId + "'");
                SqlCommand cmdUpStkQty = new SqlCommand(StrUpStkQty, con);
                cmdUpStkQty.ExecuteNonQuery();

                string StrSql = fun.select("tblInv_MaterialIssue_Details.Id", "tblInv_MaterialIssue_Master,tblInv_MaterialIssue_Details", "tblInv_MaterialIssue_Master.Id='" + MId + "' AND tblInv_MaterialIssue_Details.MId= tblInv_MaterialIssue_Master.Id AND tblInv_MaterialIssue_Master.CompId='" + CompId + "'");

                SqlCommand cmd = new SqlCommand(StrSql, con);
                SqlDataAdapter da = new SqlDataAdapter(cmd);
                DataSet DSsql = new DataSet();
                da.Fill(DSsql);

                if (DSsql.Tables[0].Rows.Count == 0)
                {
                    string delMastr = fun.delete("tblInv_MaterialIssue_Master", "CompId='" + CompId + "'  and Id='" + MId + "'");
                    SqlCommand cmdMstr = new SqlCommand(delMastr, con);
                    cmdMstr.ExecuteNonQuery();
                    con.Close();
                    Response.Redirect("MaterialIssueNote_MIN_Delete.aspx?MINNo=" + MINNo + "&ModId=9&SubModId=41");
                }
                else
                {
                    Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }               
            }
        }
      catch (Exception ex) { }
      finally 
       { con.Close(); }     
            
    }
    
    protected void BtnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("MaterialIssueNote_MIN_Delete.aspx?ModId=9&SubModId=41");
    }
    protected void GridView1_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        try
        {
            GridView1.PageIndex = e.NewPageIndex;
            this.LoadData();
        }
        catch (Exception ex) { }
              
    }
