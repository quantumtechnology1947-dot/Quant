﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_MaterialCosting_Material_Category : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    protected void Page_Load(object sender, EventArgs e)
    {
        lblMessage.Text = "";

    }
    protected void GridView1_RowUpdated(object sender, GridViewUpdatedEventArgs e)
    {
        lblMessage.Text = "Record Updated";

    }
    protected void GridView1_RowDeleted(object sender, GridViewDeletedEventArgs e)
    {
        lblMessage.Text = "Record Deleted";
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            string CDate = fun.getCurrDate();
            string CTime = fun.getCurrTime();
            string sId = Session["username"].ToString();
            string CompId = Session["compid"].ToString();
            string FinYearId = Session["finyear"].ToString();
            if (e.CommandName == "Add")
            {
                string strMaterial = ((TextBox)GridView1.FooterRow.FindControl("txtMaterial")).Text;
                string strUnit = ((DropDownList)GridView1.FooterRow.FindControl("ddUnit")).SelectedValue;
                if (strMaterial!="")
                {
                LocalSqlServer.InsertParameters["SySDate"].DefaultValue = CDate;
                LocalSqlServer.InsertParameters["SysTime"].DefaultValue = CTime;
                LocalSqlServer.InsertParameters["CompId"].DefaultValue = CompId;
                LocalSqlServer.InsertParameters["FinYearId"].DefaultValue = FinYearId;
                LocalSqlServer.InsertParameters["SessionId"].DefaultValue = sId;
                LocalSqlServer.InsertParameters["Material"].DefaultValue = strMaterial;
                LocalSqlServer.InsertParameters["Unit"].DefaultValue = strUnit;
                LocalSqlServer.Insert();
                lblMessage.Text = "Record Inserted";
                //Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }

            }

            if (e.CommandName == "Update")
            {
                
                int rowIndex = Convert.ToInt32(e.CommandArgument);
                GridViewRow row = GridView1.Rows[rowIndex];
                DropDownList ddl = (DropDownList)row.FindControl("ddUnit1");
                string value = ddl.SelectedValue;               
                string strUnit = value;
                string strMaterial = ((TextBox)row.FindControl("txtMaterial1")).Text;
                if (strMaterial != "")
                {
                    LocalSqlServer.UpdateParameters["SySDate"].DefaultValue = CDate;
                    LocalSqlServer.UpdateParameters["SysTime"].DefaultValue = CTime;
                    LocalSqlServer.UpdateParameters["CompId"].DefaultValue = CompId;
                    LocalSqlServer.UpdateParameters["FinYearId"].DefaultValue = FinYearId;
                    LocalSqlServer.UpdateParameters["SessionId"].DefaultValue = sId;
                    LocalSqlServer.UpdateParameters["Unit"].DefaultValue = strUnit;
                    LocalSqlServer.UpdateParameters["Material"].DefaultValue = strMaterial;
                    LocalSqlServer.Update();
                }
            }
            if (e.CommandName == "Add1")
            {
                string strMaterial = ((TextBox)GridView1.Controls[0].Controls[0].FindControl("txtMaterial")).Text;
                string strUnit = ((DropDownList)GridView1.Controls[0].Controls[0].FindControl("ddUnit")).SelectedValue;
                if (strMaterial != "")
                {
                    LocalSqlServer.InsertParameters["SySDate"].DefaultValue = CDate;
                    LocalSqlServer.InsertParameters["SysTime"].DefaultValue = CTime;
                    LocalSqlServer.InsertParameters["CompId"].DefaultValue = CompId;
                    LocalSqlServer.InsertParameters["FinYearId"].DefaultValue = FinYearId;
                    LocalSqlServer.InsertParameters["SessionId"].DefaultValue = sId;
                    LocalSqlServer.InsertParameters["Material"].DefaultValue = strMaterial;
                    LocalSqlServer.InsertParameters["Unit"].DefaultValue = strUnit;

                    LocalSqlServer.Insert();
                    lblMessage.Text = "Record Inserted";
                    // Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }

            }


        }
        catch(Exception ex)
        {
        }
    }



    protected void GridView1_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        try
        {
            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton edit = (LinkButton)e.Row.Cells[0].Controls[0];
                edit.Attributes.Add("onclick", "return confirmationUpdate();");

            }

            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del = (LinkButton)e.Row.Cells[1].Controls[0];
                del.Attributes.Add("onclick", "return confirmationDelete();");
            }
        }
        catch (Exception ex) { }
    }
}
