    public void LoadData()
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        con.Open();
        try
        {
            SqlCommand Cmdgrid = new SqlCommand(fun.select("tblInv_MaterialIssue_Details.Id,tblInv_MaterialIssue_Details.MRSId,tblInv_MaterialIssue_Master.MRSId as MMId,tblInv_MaterialIssue_Details.IssueQty", "tblInv_MaterialIssue_Details,tblInv_MaterialIssue_Master", "tblInv_MaterialIssue_Master.Id='" + MId + "' AND tblInv_MaterialIssue_Master.Id=tblInv_MaterialIssue_Details.MId AND tblInv_MaterialIssue_Master.CompId='" + CompId + "'"), con);
           
            SqlDataAdapter dagrid = new SqlDataAdapter(Cmdgrid);
            dagrid.Fill(DS);
            DataTable dt = new DataTable();

            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(Int32)));
            dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("PurchDesc", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ReqQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("IssueQty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ItemId", typeof(string)));

            DataRow dr;

            for (int i = 0; i < DS.Tables[0].Rows.Count; i++)
            {
                dr = dt.NewRow();
                dr[0] = Convert.ToInt32(DS.Tables[0].Rows[i]["Id"]);

                SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialRequisition_Details.ItemId,tblInv_MaterialRequisition_Details.DeptId,tblInv_MaterialRequisition_Details.WONo,tblInv_MaterialRequisition_Details.ReqQty,tblInv_MaterialRequisition_Details.Remarks", "tblInv_MaterialRequisition_Master,tblInv_MaterialRequisition_Details", "tblInv_MaterialRequisition_Details.Id='" + Convert.ToInt32(DS.Tables[0].Rows[i]["MRSId"]) + "' AND tblInv_MaterialRequisition_Master.CompId='" + CompId + "' AND tblInv_MaterialRequisition_Master.Id='" + DS.Tables[0].Rows[i]["MMId"] + "' AND tblInv_MaterialRequisition_Master.Id=tblInv_MaterialRequisition_Details.MId"), con);
                
                DataSet dsrs = new DataSet();
                SqlDataAdapter dars = new SqlDataAdapter(cmdrs);
                dars.Fill(dsrs);
                
                if (dsrs.Tables[0].Rows.Count > 0)
                {
                    SqlCommand cmditem = new SqlCommand(fun.select("ItemCode,PurchDesc,UOMPurchase", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"]) + "' AND  CompId='" + CompId + "'"), con);
                    DataSet dsitem = new DataSet();
                    SqlDataAdapter daitem = new SqlDataAdapter(cmditem);
                    daitem.Fill(dsitem);
                    if (dsitem.Tables[0].Rows.Count > 0)
                    {
                        SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOM", "Unit_Master", "Id='" + Convert.ToInt32(dsitem.Tables[0].Rows[0]["UOMPurchase"]) + "'"), con);
                        DataSet dsunit = new DataSet();
                        SqlDataAdapter daunit = new SqlDataAdapter(cmdunit);
                        daunit.Fill(dsunit);

                        dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"].ToString()));
                        dr[2] = dsitem.Tables[0].Rows[0]["PurchDesc"].ToString();
                        if (dsunit.Tables[0].Rows.Count > 0)
                        {
                            dr[3] = dsunit.Tables[0].Rows[0]["UOM"].ToString();
                        }
                        if (dsrs.Tables[0].Rows[0]["DeptId"].ToString() == "1")
                        {
                            dr[4] = "NA";
                            dr[5] = dsrs.Tables[0].Rows[0]["WONo"].ToString();
                        }
                        else
                        {
                            string dept = fun.select("Symbol", "BusinessGroup", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["DeptId"]) + "'");

                            SqlCommand cmddpt = new SqlCommand(dept, con);
                            DataSet ds = new DataSet();
                            SqlDataAdapter dadpt = new SqlDataAdapter(cmddpt);
                            dadpt.Fill(ds);
                            dr[4] = ds.Tables[0].Rows[0]["Symbol"].ToString();
                            dr[5] = "NA";
                        }

                        dr[6] =Convert.ToDouble(Decimal.Parse(( dsrs.Tables[0].Rows[0]["ReqQty"]).ToString()).ToString("N3"));
                        dr[7] =Convert.ToDouble( Decimal.Parse((DS.Tables[0].Rows[i]["IssueQty"]).ToString()).ToString("N3"));
                        dr[8] = dsrs.Tables[0].Rows[0]["Remarks"].ToString();
                        dr[9] = Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"]);
                    }
                }

                dt.Rows.Add(dr);
                dt.AcceptChanges();

            }

            GridView1.DataSource = dt;
            GridView1.DataBind();
        }
        catch (Exception ex) { }
        finally { }

    }

    protected void GridView1_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            double StkQty = 0;
            double BalQty = 0;
            int index = GridView1.EditIndex;
            string CDate = fun.getCurrDate();
            string CTime = fun.getCurrTime();
            GridViewRow row = GridView1.Rows[index];
            int id = Convert.ToInt32(((Label)row.FindControl("lblId")).Text);
            int ItemId = Convert.ToInt32(((Label)row.FindControl("lblItemId")).Text);
            double IssueQty = Convert.ToDouble(decimal.Parse((((Label)row.FindControl("lblIssueQty1")).Text).ToString()).ToString("N3"));
            double Qty = Convert.ToDouble(decimal.Parse((((TextBox)row.FindControl("txtAccQty")).Text).ToString()).ToString("N3"));
            
            con.Open();

            if (IssueQty >= Qty && fun.NumberValidationQty(Qty.ToString()) == true)
            {
                string StrIcode = fun.select("StockQty", "tblDG_Item_Master", "Id='" + ItemId + "' AND CompId='" + CompId + "'");
                SqlCommand cmdIcode = new SqlCommand(StrIcode, con);
                SqlDataAdapter daIcode = new SqlDataAdapter(cmdIcode);
                DataSet DSIcode = new DataSet();
                daIcode.Fill(DSIcode);
                // For ItemCode
                if (DSIcode.Tables[0].Rows.Count > 0)
                {
                    StkQty = Convert.ToDouble(decimal.Parse((DSIcode.Tables[0].Rows[0]["StockQty"]).ToString()).ToString("N3"));
                }
                BalQty = (IssueQty - Qty) + StkQty;
                
                string sqlupBalQty = fun.update("tblDG_Item_Master", "StockQty='" + BalQty + "'", "CompId='" + CompId + "' AND Id='" + ItemId + "'");
                SqlCommand cmdBalQty = new SqlCommand(sqlupBalQty, con);
                cmdBalQty.ExecuteNonQuery();
               
                string sql = fun.update("tblInv_MaterialIssue_Details", "IssueQty='" + Qty + "'", "Id='" + id + "' And MId='" + MId + "'");
                SqlCommand cmd = new SqlCommand(sql, con);
                cmd.ExecuteNonQuery();
               
                string sql2 = fun.update("tblInv_MaterialIssue_Master", "SysDate='" + CDate + "',SysTime='" + CTime + "',SessionId='" + sId + "'", "Id='" + MId + "' And CompId='" + CompId + "'");
                SqlCommand cmd2 = new SqlCommand(sql2, con);
                cmd2.ExecuteNonQuery();
                con.Close();
                GridView1.EditIndex = -1;
                this.LoadData();
            }
        }
        catch (Exception ex) { }

    }
