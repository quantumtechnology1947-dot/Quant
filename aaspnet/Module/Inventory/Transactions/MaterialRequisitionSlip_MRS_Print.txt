﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_Inventory_Transactions_MaterialRequisitionSlip_MRS_Print : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    DataSet DS = new DataSet();
   
    int CompId = 0;
    string mrs = "";
    string emp = "";
    int FinYearId = 0;

    protected void Page_Load(object sender, EventArgs e)
    {
        FinYearId= Convert.ToInt32(Session["finyear"]);
        CompId = Convert.ToInt32(Session["compid"]);

        if (!IsPostBack)
        {
            txtMrsNo.Visible = false;
            this.loadData();
        }
    }


    public void loadData()
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        con.Open();
        try
        {
            string x = "";

            if (drpfield.SelectedValue == "1")
            {
                if (txtMrsNo.Text != "")
                {
                    x = " AND MRSNo='" + txtMrsNo.Text + "'";
                }
            }

            string y = "";
            if (drpfield.SelectedValue == "0")
            {
                if (txtEmpName.Text != "")
                {
                    y = " AND SessionId='" + fun.getCode(txtEmpName.Text) + "'";
                }
            }

            SqlDataAdapter da = new SqlDataAdapter("Sp_MRS_FillGrid", con);
            da.SelectCommand.CommandType = CommandType.StoredProcedure;
            da.SelectCommand.Parameters.Add(new SqlParameter("@CompId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@CompId"].Value = CompId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@FinId", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@FinId"].Value = FinYearId;
            da.SelectCommand.Parameters.Add(new SqlParameter("@x", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@x"].Value = x;
            da.SelectCommand.Parameters.Add(new SqlParameter("@y", SqlDbType.VarChar));
            da.SelectCommand.Parameters["@y"].Value = y;
            DataSet DSitem = new DataSet();
            da.Fill(DSitem);
            GridView1.DataSource = DSitem;
            GridView1.DataBind();

        }
        catch (Exception ex) { }
        finally
        {
            con.Close();
        }

    }
    protected void GridView1_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        try
        {
            GridView1.PageIndex = e.NewPageIndex;
            this.loadData();

        }
        catch (Exception ex)
        {
        }
    }

    protected void GridView1_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {
            if (e.CommandName == "Sel")
            {
                GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);

                string MRSNo = ((Label)row.FindControl("lblMRSNo")).Text;
                string fyid = ((Label)row.FindControl("lblFinId")).Text;
                string Id = ((Label)row.FindControl("Id")).Text;
                string getRandomKey = fun.GetRandomAlphaNumeric();
                Response.Redirect("MaterialRequisitionSlip_MRS_Print_Details.aspx?Id=" + Id + "&MRSNo=" + MRSNo + "&FyId=" + fyid + "&Key=" + getRandomKey + "&ModId=9&SubModId=40");
            }
        }
        catch (Exception ex) { }

    }
    protected void Button1_Click(object sender, EventArgs e)
    {

        this.loadData();
    }

    protected void drpfield_SelectedIndexChanged(object sender, EventArgs e)
    {
        try
        {
            if (drpfield.SelectedValue == "1" )
            {
                txtMrsNo.Visible = true;
                txtMrsNo.Text = "";
                txtEmpName.Visible = false;
                this.loadData();
            }
            else
            {
                txtMrsNo.Visible = false;
                txtEmpName.Visible = true;
                txtEmpName.Text = "";
                this.loadData();
            }
        }
        catch (Exception ex) { }
    }
    [System.Web.Services.WebMethodAttribute(), System.Web.Script.Services.ScriptMethodAttribute()]
    public static string[] GetCompletionList(string prefixText, int count, string contextKey)
    {
        clsFunctions fun = new clsFunctions();
        string connStr = fun.Connection();
        DataSet ds = new DataSet();
        SqlConnection con = new SqlConnection(connStr);
        int CompId = Convert.ToInt32(HttpContext.Current.Session["compid"]);
        con.Open();
        string cmdStr = fun.select("EmpId,EmployeeName", "tblHR_OfficeStaff","CompId='"+CompId+"'");
        SqlDataAdapter da = new SqlDataAdapter(cmdStr, con);
        da.Fill(ds, "tblHR_OfficeStaff");
        con.Close();
        string[] main = new string[0];
        for (int i = 0; i < ds.Tables[0].Rows.Count; i++)
        {
            if (ds.Tables[0].Rows[i].ItemArray[1].ToString().ToLower().StartsWith(prefixText.ToLower()))
            {
                Array.Resize(ref main, main.Length + 1);
                main[main.Length - 1] = ds.Tables[0].Rows[i].ItemArray[1].ToString() + " [" + ds.Tables[0].Rows[i].ItemArray[0].ToString() + "]";
                //if (main.Length == 10)
                //    break;
            }
        }
        Array.Sort(main);
        return main;
    }
}
