﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;

using System.Globalization;

public partial class Module_HR_Transactions_Salary_Delete_Details : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    int CompId = 0;
    int FinYearId = 0;
    string EmpId = "";

    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {

            EmpId = Request.QueryString["EmpId"].ToString();
            CompId = Convert.ToInt32(Session["compid"]);
            FinYearId = Convert.ToInt32(Session["finyear"]);

            if (!IsPostBack)
            {
                this.binddata();
            }
        }
       catch (Exception ex) { }
    }

    public void binddata()
    {
        string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            con.Open();

       try
        {           

            DataTable dt = new DataTable();
            DataRow dr;

            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("Month", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("DaysOfMonth", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Holidays", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MonthSunday", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WorkingDays", typeof(string)));

            dt.Columns.Add(new System.Data.DataColumn("Present", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Absent", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("LateIn", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("HalfDay", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Sunday", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Coff", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("PL", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("OverTimeHrs", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("OverTimeRate", typeof(string)));

            dt.Columns.Add(new System.Data.DataColumn("BankLoan", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Installment", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MobileExeAmt", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Addition", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Deduction", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MId", typeof(int)));
            string StrLeave = fun.select("tblHR_Salary_Details.Id,tblHR_Salary_Details.MId,tblHR_Salary_Details.Present,tblHR_Salary_Details.Absent,tblHR_Salary_Details.LateIn,tblHR_Salary_Details.HalfDay,tblHR_Salary_Details.Sunday,tblHR_Salary_Details.Coff,tblHR_Salary_Details.PL,tblHR_Salary_Details.OverTimeHrs,tblHR_Salary_Details.OverTimeRate,tblHR_Salary_Details.Installment,tblHR_Salary_Details.MobileExeAmt,tblHR_Salary_Details.Addition,tblHR_Salary_Details.Deduction,tblHR_Salary_Master.FMonth", "tblHR_Salary_Details,tblHR_Salary_Master", "tblHR_Salary_Details.MId=tblHR_Salary_Master.Id AND tblHR_Salary_Master.CompId='" + CompId + "' AND tblHR_Salary_Master.FinYearId='" + FinYearId + "' AND tblHR_Salary_Master.EmpId='" + EmpId + "'");
            SqlCommand cmdLeave = new SqlCommand(StrLeave, con);
            //SqlDataAdapter DALeave = new SqlDataAdapter(cmdLeave);
            //DataSet DSLeave = new DataSet();
            //DALeave.Fill(DSLeave);
            SqlDataReader DSLeave = cmdLeave.ExecuteReader();

            //for (int i = 0; i < DSLeave.Tables[0].Rows.Count; i++)
            while (DSLeave.Read())
            {
                int g = Convert.ToInt32(DSLeave["FMonth"]);
                int g1 = fun.SalYrs(FinYearId, g, CompId);

                dr = dt.NewRow();

                dr[0] = DSLeave["Id"].ToString();

                dr[1] = CultureInfo.CurrentCulture.DateTimeFormat.GetMonthName(Convert.ToInt32(DSLeave["FMonth"]));
                dr[2] = DateTime.DaysInMonth(g1, g);
                dr[3] = fun.GetHoliday(g, CompId, FinYearId).ToString();
                dr[4] = fun.CountSundays(g1, g);
                dr[5] = fun.WorkingDays(FinYearId, g).ToString();
                dr[6] = DSLeave["Present"].ToString();
                dr[7] = DSLeave["Absent"].ToString();
                dr[8] = DSLeave["LateIn"].ToString();
                dr[9] = DSLeave["HalfDay"].ToString();
                dr[10] = DSLeave["Sunday"].ToString();
                dr[11] = DSLeave["Coff"].ToString();
                dr[12] = DSLeave["PL"].ToString();
                dr[13] = DSLeave["OverTimeHrs"].ToString();
                dr[14] = DSLeave["OverTimeRate"].ToString();

                double LoanAmt = 0;
                string StrBankLoan = fun.select("Sum(Amount) as LoanAmt", "tblHR_BankLoan", "CompId='" + CompId + "' And FinYearId<='" + FinYearId + "' And EmpId='" + EmpId + "'");
                SqlCommand cmdBankLoan = new SqlCommand(StrBankLoan, con);
                //SqlDataAdapter daBankLoan = new SqlDataAdapter(cmdBankLoan);
                //DataSet dsBankLoan = new DataSet();
                //daBankLoan.Fill(dsBankLoan);
                SqlDataReader dsBankLoan = cmdBankLoan.ExecuteReader();
                dsBankLoan.Read();

                while (dsBankLoan.Read())
                {
                    if (Convert.ToDouble(dsBankLoan["LoanAmt"]) > 0 && dsBankLoan["LoanAmt"] != DBNull.Value)
                    {
                        LoanAmt = Convert.ToDouble(dsBankLoan["LoanAmt"]);
                    }
                }

                dr[15] = LoanAmt;
                dr[16] = DSLeave["Installment"].ToString();
                dr[17] = DSLeave["MobileExeAmt"].ToString();
                dr[18] = DSLeave["Addition"].ToString();
                dr[19] = DSLeave["Deduction"].ToString();
                dr[20] = DSLeave["MId"].ToString();
                dt.Rows.Add(dr);
                dt.AcceptChanges();
            }

            GridView2.DataSource = dt;
            GridView2.DataBind();
        }
      catch (Exception ex) { }
      finally
        {
            con.Close();
        }
    }

    protected void GridView2_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        try
        {
            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del = (LinkButton)e.Row.Cells[0].Controls[0];
                del.Attributes.Add("onclick", "return confirmationDelete();");
            }
        }
        catch (Exception ex)
        {
        }


    }
    protected void GridView2_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        try
        {
            GridView2.PageIndex = e.NewPageIndex;
            this.binddata();
        }
        catch (Exception st)
        {

        }
    }

    protected void GridView2_RowCommand(object sender, GridViewCommandEventArgs e)
    {

        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        con.Open();

        try
        {
            if (e.CommandName == "Del")
            {
                GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
                string MId = ((Label)row.FindControl("lblMId")).Text;

                string sql = fun.delete("tblHR_Salary_Details", "MId='" + MId + "'");
                SqlCommand cmd = new SqlCommand(sql, con);
                cmd.ExecuteNonQuery();

                string sql11 = fun.delete("tblHR_Salary_Master", "Id='" + MId + "'");
                SqlCommand cmd11 = new SqlCommand(sql11, con);
                cmd11.ExecuteNonQuery();
                this.binddata();
            }
        }
        catch (Exception st)
        {
        }

        con.Close();
    }


    protected void btnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("Salary_Delete.aspx?ModId=12&SubModId=133");
    }
}
