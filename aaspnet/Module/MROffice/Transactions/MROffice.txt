﻿using System;
using System.Collections;
using System.Configuration;

public partial class Module_MROffice_Transactions_MROffice : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    SqlConnection con;

    string CompId = string.Empty;
    string FinYearId = string.Empty;
    string SessionId = string.Empty;
    string constr = string.Empty;
    string CurrDate = string.Empty;
    string CurrTime = string.Empty;

    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            constr = fun.Connection();
            con = new SqlConnection(constr);
            CompId = Session["compid"].ToString();
            FinYearId = Session["finyear"].ToString();
            SessionId = Session["username"].ToString();

            if (!IsPostBack)
            {
                this.BindData();
            }
        }
        catch (Exception sx)
        {

        }
    }

    public void BindData()
    {
       try
        {

        DataTable dt = new DataTable();

        dt.Columns.Add("Id", typeof(int));
        dt.Columns.Add("ForModule", typeof(string));
        dt.Columns.Add("Format", typeof(string));
        dt.Columns.Add("FileName", typeof(string));

        string StrSql = fun.select("*", "tblMROffice", "CompId='" + CompId + "' Order by ForModule ASC");
        SqlCommand cmd = new SqlCommand(StrSql, con);
        SqlDataAdapter da = new SqlDataAdapter(cmd);
        DataSet ds = new DataSet();
        da.Fill(ds);

        DataRow dr;

        for (int i = 0; i < ds.Tables[0].Rows.Count; i++)
        {
            dr = dt.NewRow();
            dr[0] = ds.Tables[0].Rows[i]["Id"].ToString();

            string StrSql2 = fun.select("*", "tblModule_Master", "ModId='" + ds.Tables[0].Rows[i]["ForModule"].ToString() + "'");
            SqlCommand cmd2 = new SqlCommand(StrSql2, con);
            SqlDataAdapter da2 = new SqlDataAdapter(cmd2);
            DataSet ds2 = new DataSet();
            da2.Fill(ds2);

            dr[1] = ds2.Tables[0].Rows[0]["ModName"].ToString();
            dr[2] = ds.Tables[0].Rows[i]["Format"].ToString();

            if (ds.Tables[0].Rows[i]["FileName"].ToString() != "" && ds.Tables[0].Rows[i]["FileName"] != DBNull.Value)
            {
                dr[3] = ds.Tables[0].Rows[i]["FileName"].ToString();
            }

            dt.Rows.Add(dr);
            dt.AcceptChanges();
        }

        GridView2.DataSource = dt;
        GridView2.DataBind();

         }
        catch (Exception sx)
        {

        }
    }

   

    protected void GridView2_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {
            CurrDate = fun.getCurrDate();
            CurrTime = fun.getCurrTime();
            con.Open();

            if (e.CommandName == "Add")
            {
                int ForModule = Convert.ToInt32(((DropDownList)GridView2.FooterRow.FindControl("drpFor")).SelectedValue);
                string Format = ((TextBox)GridView2.FooterRow.FindControl("txtFormat")).Text;
                string strcv = "";

                HttpPostedFile mycv = ((FileUpload)GridView2.FooterRow.FindControl("FileUploadAtt")).PostedFile;
                Byte[] mycvdata = null;

                if (((FileUpload)GridView2.FooterRow.FindControl("FileUploadAtt")) != null)
                {
                    Stream fscv = ((FileUpload)GridView2.FooterRow.FindControl("FileUploadAtt")).PostedFile.InputStream;
                    BinaryReader brcv = new BinaryReader(fscv);
                    mycvdata = brcv.ReadBytes((Int32)fscv.Length);
                    strcv = Path.GetFileName(mycv.FileName);
                }

                string cmdstr = fun.insert("tblMROffice", "SysDate,SysTime,CompId,SessionId,FinYearId,ForModule,Format,FileName,Size,ContentType,Data", "'" + CurrDate + "','" + CurrTime + "','" + CompId + "','" + SessionId + "','" + FinYearId + "','" + ForModule + "','" + Format + "','" + strcv + "','" + mycvdata.Length + "','" + mycv.ContentType + "',@Data");
                SqlCommand cmd = new SqlCommand(cmdstr, con);
                cmd.Parameters.AddWithValue("@Data", mycvdata);
                cmd.ExecuteNonQuery();
                this.BindData();
            }

            if (e.CommandName == "AddEmpty")
            {
                int ForModule = Convert.ToInt32(((DropDownList)GridView2.Controls[0].Controls[0].FindControl("drpFor")).SelectedValue);
                string Format = ((TextBox)GridView2.Controls[0].Controls[0].FindControl("txtFormat")).Text;
                string strcv = "";

                HttpPostedFile mycv = ((FileUpload)GridView2.Controls[0].Controls[0].FindControl("FileUploadAtt")).PostedFile;
                Byte[] mycvdata = null;

                if (((FileUpload)GridView2.Controls[0].Controls[0].FindControl("FileUploadAtt")) != null)
                {
                    Stream fscv = ((FileUpload)GridView2.Controls[0].Controls[0].FindControl("FileUploadAtt")).PostedFile.InputStream;
                    BinaryReader brcv = new BinaryReader(fscv);
                    mycvdata = brcv.ReadBytes((Int32)fscv.Length);
                    strcv = Path.GetFileName(mycv.FileName);
                }

                string cmdstr = fun.insert("tblMROffice", "SysDate,SysTime,CompId,SessionId,FinYearId,ForModule,Format,FileName,Size,ContentType,Data", "'" + CurrDate + "','" + CurrTime + "','" + CompId + "','" + SessionId + "','" + FinYearId + "','" + ForModule + "','" + Format + "','" + strcv + "','" + mycvdata.Length + "','" + mycv.ContentType + "',@Data");
                SqlCommand cmd = new SqlCommand(cmdstr, con);
                cmd.Parameters.AddWithValue("@Data", mycvdata);
                cmd.ExecuteNonQuery();
                this.BindData();
            }

            if (e.CommandName == "Download")
            {
                GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
                int id = Convert.ToInt32(((HiddenField)row.FindControl("hidId")).Value);

                Response.Redirect("~/Controls/DownloadFile.aspx?Id=" + id + "&tbl=tblMROffice&qfd=Data&qfn=FileName&qct=ContentType");

            }

            if (e.CommandName == "Delete")
            {
                GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
                int id = Convert.ToInt32(((HiddenField)row.FindControl("hidId")).Value);
                string cmdstr4 = fun.delete("tblMROffice", "Id='" + id + "'");
                Response.Write(cmdstr4);
                SqlCommand cmd = new SqlCommand(cmdstr4, con);
                cmd.ExecuteNonQuery();
                this.BindData();
            }

            con.Close();
            //Page.Response.Redirect(Page.Request.Url.ToString(), true);

        }
        
        catch(Exception ett)
        {

        }

    }


    
    protected void GridView2_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        try
        {
            GridView2.PageIndex = e.NewPageIndex;
            this.BindData();
            
        }
        catch (Exception sx)
        {

        }
    }
}
