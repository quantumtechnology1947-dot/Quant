﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;

public partial class Module_Design_Masters_ECNReasonTypes : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    protected void Page_Load(object sender, EventArgs e)
    {
        lblmsg.Text = "";
    }

    protected void GridView2_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        try
        {
            string sysdate = fun.getCurrDate();
            string systime = fun.getCurrTime();
            string compid = Session["compid"].ToString();            

            if (e.CommandName == "Add")
            {
                string strType = ((TextBox)GridView2.FooterRow.FindControl("txtTypes")).Text;
                SqlDataSource1.InsertParameters["Types"].DefaultValue = strType;
                SqlDataSource1.InsertParameters["compid"].DefaultValue = compid;
               
                if (strType != "")
                {
                    SqlDataSource1.Insert();
                    Page.Response.Redirect(Page.Request.Url.ToString(),true);
                }

            }

            if (e.CommandName == "Update")
            {
                string strType = ((TextBox)GridView2.FooterRow.FindControl("txtTypes")).Text;
                SqlDataSource1.UpdateParameters["compid"].DefaultValue = compid;
                SqlDataSource1.UpdateParameters["Types"].DefaultValue = strType;
                SqlDataSource1.Update();
                Page.Response.Redirect(Page.Request.Url.ToString(), true);
            }

            if (e.CommandName == "Add1")
            {
                string strType = ((TextBox)GridView2.Controls[0].Controls[0].FindControl("txtTypes0")).Text;
                SqlDataSource1.InsertParameters["Types"].DefaultValue = strType;
                SqlDataSource1.InsertParameters["compid"].DefaultValue = compid;
                if (strType != "")
                {
                    SqlDataSource1.Insert();
                    Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }
            }


        }
        catch (Exception ex)
        {

        }
    }

    protected void GridView2_RowUpdated(object sender, GridViewUpdatedEventArgs e)
    {
        lblmsg.Text = "Record Updated.";
    }

    protected void GridView2_RowDeleted(object sender, GridViewDeletedEventArgs e)
    {
        lblmsg.Text = "Record Deleted.";
    }

    protected void GridView2_RowDataBound(object sender, GridViewRowEventArgs e)
    {
        try
        {
            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton upd = (LinkButton)e.Row.Cells[1].Controls[0];
                upd.Attributes.Add("onclick", "return confirmationUpdate();");
            }

            if (e.Row.RowType == DataControlRowType.DataRow)
            {
                LinkButton del = (LinkButton)e.Row.Cells[2].Controls[0];
                del.Attributes.Add("onclick", "return confirmationDelete();");
            }

        }
        catch (Exception ex)
        {
        }
    }

}
