    public void loadData(string mrqn, string empid)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        con.Open();

        try
        {

            string x = string.Empty;

            if (drpfield.SelectedValue == "0")
            {
                if (txtMqnNo.Text != "")
                {
                    x = " AND MRQNNo='" + txtMqnNo.Text + "'";
                }
            }

            if (drpfield.SelectedValue == "Select")
            {
                txtMqnNo.Visible = true;
                txtEmpName.Visible = false;
            }

            string p = string.Empty;
            if (drpfield.SelectedValue == "1")
            {
                if (txtMqnNo.Text != "")
                {
                    p = " AND MRNNo='" + txtMqnNo.Text + "'";
                }
            }

            string y = string.Empty;
            if (drpfield.SelectedValue == "2")
            {
                if (txtEmpName.Text != "")
                {
                    y = " AND SessionId='" + fun.getCode(txtEmpName.Text) + "'";
                }
            }
            string z = string.Empty;
            if (drpfield.SelectedValue == "3")
            {
                if (txtMqnNo.Text != "")
                {
                    z = " AND ScrapNo='" + txtMqnNo.Text + "'";
                }
            }

            string StrSC = fun.select(" Distinct(ScrapNo ),SysDate ,CompId  ,FinYearId  ,SessionId  ,MRQNId ", "tblQC_Scrapregister", "FinYearId<='" + FinYearId + "' And CompId='" + CompId + "' " + z);
            SqlCommand sqlsc = new SqlCommand(StrSC, con);
            SqlDataReader dssc = sqlsc.ExecuteReader();

            DataTable dt = new DataTable();
            DataRow dr;

            dt.Columns.Add(new System.Data.DataColumn("FinYearId", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("FinYear", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MRQNDATE", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MRQNNo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MRNNo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("GenBy", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ScrapNo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("SysDate", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("MRNDate", typeof(string)));

            while (dssc.Read())
            {
                dr = dt.NewRow();

                SqlCommand Cmdgrid = new SqlCommand(fun.select("SysDate As MRQNDATE,MRNNo,MRNId,MRQNNo,FinYearId,SessionId", "tblQc_MaterialReturnQuality_Master ", " Id='" + dssc["MRQNId"].ToString() + "'" + x + p + y), con);
                SqlDataReader DS = Cmdgrid.ExecuteReader();
                DS.Read();

                if (DS.HasRows == true)
                {
                    dr[4] = DS["MRNNo"].ToString();
                    dr[3] = DS["MRQNNo"].ToString();
                    dr[2] = fun.FromDateDMY(DS["MRQNDATE"].ToString());

                    SqlCommand Cmdgrid1 = new SqlCommand(fun.select("Id,SysDate As MRNDATE", "tblInv_MaterialReturn_Master ", "  Id='" + DS["MRNId"].ToString() + "' "), con);
                    SqlDataReader DS1 = Cmdgrid1.ExecuteReader();
                    DS1.Read();

                    if (DS1.HasRows == true)
                    {
                        dr[8] = fun.FromDateDMY(DS1["MRNDate"].ToString());
                    }

                    int FinYr = Convert.ToInt32(dssc["FinYearId"]);

                    string sqlFin = fun.select("FinYear,FinYearId", "tblFinancial_master", "FinYearId='" + FinYr + "' AND CompId='" + CompId + "'");
                    SqlCommand cmdFinYr = new SqlCommand(sqlFin, con);
                    SqlDataReader DSFin = cmdFinYr.ExecuteReader();
                    DSFin.Read();

                    string sql2 = fun.select("Title+'. '+EmployeeName As EmpName", "tblHR_OfficeStaff", "CompId='" + CompId + "' AND EmpId='" + dssc["SessionId"] + "'");
                    SqlCommand cmd2 = new SqlCommand(sql2, con);
                    SqlDataReader DS2 = cmd2.ExecuteReader();
                    DS2.Read();

                    if (DSFin.HasRows == true)
                    {
                        dr[0] = DSFin["FinYearId"].ToString();
                        dr[1] = DSFin["FinYear"].ToString();
                    }

                    if (DS2.HasRows == true)
                    {
                        dr[5] = DS2["EmpName"].ToString();
                    }

                    if (dssc.HasRows == true)
                    {
                        dr[6] = dssc["ScrapNo"].ToString();
                        dr[7] = fun.FromDateDMY(dssc["SysDate"].ToString());
                    }

                    dt.Rows.Add(dr);
                    dt.AcceptChanges();
                }
            }
            GridView1.DataSource = dt;
            GridView1.DataBind();
        }
        catch (Exception ex) { }
        finally
        { con.Close(); }
    }

    [System.Web.Services.WebMethodAttribute(), System.Web.Script.Services.ScriptMethodAttribute()]
    public static string[] GetCompletionList(string prefixText, int count, string contextKey)
    {
        clsFunctions fun = new clsFunctions();
        string connStr = fun.Connection();
        DataSet ds = new DataSet();
        SqlConnection con = new SqlConnection(connStr);
        int CompId = Convert.ToInt32(HttpContext.Current.Session["compid"]);
        con.Open();
        string cmdStr = fun.select("EmpId,EmployeeName", "tblHR_OfficeStaff", "CompId='" + CompId + "'");
        SqlDataAdapter da = new SqlDataAdapter(cmdStr, con);
        da.Fill(ds, "tblHR_OfficeStaff");
        con.Close();
        string[] main = new string[0];
        for (int i = 0; i < ds.Tables[0].Rows.Count; i++)
        {
            if (ds.Tables[0].Rows[i].ItemArray[1].ToString().ToLower().StartsWith(prefixText.ToLower()))
            {
                Array.Resize(ref main, main.Length + 1);
                main[main.Length - 1] = ds.Tables[0].Rows[i].ItemArray[1].ToString() + " [" + ds.Tables[0].Rows[i].ItemArray[0].ToString() + "]";
                //if (main.Length == 10)
                //    break;
            }
        }
        Array.Sort(main);
        return main;
    }
