﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_SalesDistribution_Transactions_CustEnquiry_Dashboard : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();

    protected void Page_Load(object sender, EventArgs e)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);

        try
        {
            con.Open();
            int CompId = Convert.ToInt32(Session["compid"]);
            string SId = Session["username"].ToString();
            int FinYear = Convert.ToInt32(Session["finyear"]);

            string DSBN = fun.delete("tblFile_Attachment", "CompId ='" + CompId + "' AND FinYearId ='" + FinYear + "' AND SessionId = '" + SId + "'");
            SqlCommand commandde = new SqlCommand(DSBN, con);
            commandde.ExecuteNonQuery();
        }
        catch (Exception ex)
        {

        }
        finally
        {
            con.Close();  
        }
    }
}
