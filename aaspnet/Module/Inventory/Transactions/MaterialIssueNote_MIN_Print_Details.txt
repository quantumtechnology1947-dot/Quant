﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;
using CrystalDecisions.CrystalReports.Engine;

public partial class Module_Inventory_Transactions_MaterialIssueNote_MIN_Print_Details : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    DataSet DS = new DataSet();
    string MINNo = "";
    string MRSNo = "";
    int FyId = 0;
    int CompId = 0;
    ReportDocument cryRpt = new ReportDocument();
    string MId = "";
    string Key = string.Empty;
    protected void Page_Init(object sender, EventArgs e)
    {

        Key = Request.QueryString["Key"].ToString();
        if (!IsPostBack)
        {

            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            con.Open();
            DataTable dt = new DataTable(); 
            DataSet Min = new DataSet();
            try
            {
                MId = Request.QueryString["Id"].ToString();
                MINNo = Request.QueryString["MINNo"].ToString();
                MRSNo = Request.QueryString["MRSNo"].ToString();
                FyId = Convert.ToInt32(Request.QueryString["FYId"]);
                CompId = Convert.ToInt32(Session["compid"]);

                SqlCommand Cmdgrid = new SqlCommand(fun.select("tblInv_MaterialIssue_Details.Id,tblInv_MaterialIssue_Details.MRSId,tblInv_MaterialIssue_Master.MRSId as MMId,tblInv_MaterialIssue_Details.IssueQty", "tblInv_MaterialIssue_Details,tblInv_MaterialIssue_Master", "tblInv_MaterialIssue_Master.Id='" + MId + "' AND tblInv_MaterialIssue_Master.Id=tblInv_MaterialIssue_Details.MId AND tblInv_MaterialIssue_Master.CompId='" + CompId + "'"), con);
                SqlDataReader rdr = Cmdgrid.ExecuteReader();
                dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
                dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("ManfDesc", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("Symbol", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("ReqQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("IssueQty", typeof(double)));
                dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
                dt.Columns.Add(new System.Data.DataColumn("CompId", typeof(int)));

               
                DataRow dr;
                //for (int i = 0; i < DS.Tables[0].Rows.Count; i++)
                while(rdr.Read())
                {


                    dr = dt.NewRow();

                    dr[0] = Convert.ToInt32(rdr["Id"]);

                    SqlCommand cmdrs = new SqlCommand(fun.select("tblInv_MaterialRequisition_Details.ItemId,tblInv_MaterialRequisition_Details.DeptId,tblInv_MaterialRequisition_Details.WONo,tblInv_MaterialRequisition_Details.ReqQty,tblInv_MaterialRequisition_Details.Remarks", "tblInv_MaterialRequisition_Master,tblInv_MaterialRequisition_Details", "tblInv_MaterialRequisition_Details.Id='" + Convert.ToInt32(rdr["MRSId"]) + "' AND tblInv_MaterialRequisition_Master.CompId='" + CompId + "' AND tblInv_MaterialRequisition_Master.Id='" + rdr["MMId"] + "' AND tblInv_MaterialRequisition_Master.Id=tblInv_MaterialRequisition_Details.MId"), con);

                    DataSet dsrs = new DataSet();
                    SqlDataAdapter dars = new SqlDataAdapter(cmdrs);

                    dars.Fill(dsrs);
                    if (dsrs.Tables[0].Rows.Count > 0)
                    {
                        SqlCommand cmditem = new SqlCommand(fun.select("ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"]) + "' AND  CompId='" + CompId + "'"), con);
                        DataSet dsitem = new DataSet();
                        SqlDataAdapter daitem = new SqlDataAdapter(cmditem);
                        daitem.Fill(dsitem);
                        if (dsitem.Tables[0].Rows.Count > 0)
                        {
                            SqlCommand cmdunit = new SqlCommand(fun.select("Symbol As UOM", "Unit_Master", "Id='" + Convert.ToInt32(dsitem.Tables[0].Rows[0]["UOMBasic"]) + "'"), con);
                            DataSet dsunit = new DataSet();
                            SqlDataAdapter daunit = new SqlDataAdapter(cmdunit);
                            daunit.Fill(dsunit);
                            if (dsunit.Tables[0].Rows.Count > 0)
                            {
                                dr[3] = dsunit.Tables[0].Rows[0]["UOM"].ToString();
                            }
                            dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(dsrs.Tables[0].Rows[0]["ItemId"].ToString()));
                            dr[2] = dsitem.Tables[0].Rows[0]["ManfDesc"].ToString();

                        }
                        if (dsrs.Tables[0].Rows[0]["DeptId"].ToString() == "1")
                        {
                            dr[4] = "NA";
                            dr[5] = dsrs.Tables[0].Rows[0]["WONo"].ToString();
                        }
                        else
                        {
                            if (dsrs.Tables[0].Rows[0]["DeptId"] != DBNull.Value)
                            {


                                string dept = fun.select("Symbol", "BusinessGroup", "Id='" + Convert.ToInt32(dsrs.Tables[0].Rows[0]["DeptId"]) + "'");

                                SqlCommand cmddpt = new SqlCommand(dept, con);
                                DataSet ds = new DataSet();
                                SqlDataAdapter dadpt = new SqlDataAdapter(cmddpt);
                                dadpt.Fill(ds);

                                if (ds.Tables[0].Rows.Count > 0)
                                {
                                    dr[4] = ds.Tables[0].Rows[0]["Symbol"].ToString();
                                    dr[5] = "NA";
                                }
                            }
                            else
                            {
                                dr[5] = dsrs.Tables[0].Rows[0]["WONo"].ToString();
                                dr[4] = "NA";
                            }

                        }
                        dr[6] = Convert.ToDouble(decimal.Parse((dsrs.Tables[0].Rows[0]["ReqQty"]).ToString()).ToString("N3"));
                        dr[7] = Convert.ToDouble(decimal.Parse((rdr["IssueQty"]).ToString()).ToString("N3"));
                        dr[8] = dsrs.Tables[0].Rows[0]["Remarks"].ToString();
                        dr[9] = CompId;
                    }
                    dt.Rows.Add(dr);
                    dt.AcceptChanges();
                }

                Min.Tables.Add(dt);
                DataSet xsdds = new MIN();
                xsdds.Tables[0].Merge(Min.Tables[0]);
                xsdds.AcceptChanges();
                cryRpt = new ReportDocument();
                cryRpt.Load(Server.MapPath("~/Module/Inventory/Transactions/Reports/MIN_Print.rpt"));
                cryRpt.SetDataSource(xsdds);

                // For MIN Date
                string sqlMINDate = fun.select("SessionId,SysDate", "tblInv_MaterialIssue_Master", "Id='" + MId + "' AND CompId='" + CompId + "' ");
                SqlCommand cmdMINDate = new SqlCommand(sqlMINDate, con);
                SqlDataAdapter daMINDate = new SqlDataAdapter(cmdMINDate);
                DataSet DSMINDate = new DataSet();
                daMINDate.Fill(DSMINDate);
                if (DSMINDate.Tables[0].Rows.Count > 0)
                {
                    string MINDate = fun.FromDateDMY(DSMINDate.Tables[0].Rows[0]["SysDate"].ToString());
                    cryRpt.SetParameterValue("MINDate", MINDate);
                }
                // For Gen By.
                string sqlGenBy = fun.select("Title+'. '+EmployeeName As EmpName", "tblHR_OfficeStaff", "CompId='" + CompId + "' AND EmpId='" + DSMINDate.Tables[0].Rows[0]["SessionId"] + "'");
                SqlCommand cmdGenBy = new SqlCommand(sqlGenBy, con);
                SqlDataAdapter daGenBy = new SqlDataAdapter(cmdGenBy);
                DataSet DSGenBy = new DataSet();
                daGenBy.Fill(DSGenBy);
                if (DSGenBy.Tables[0].Rows.Count > 0)
                {
                    string GenBy = DSGenBy.Tables[0].Rows[0]["EmpName"].ToString();
                    cryRpt.SetParameterValue("GenBy", GenBy);
                }
                // For MIN NO.
                cryRpt.SetParameterValue("MINNo", MINNo);
                // For MRS NO.
                cryRpt.SetParameterValue("MRSNo", MRSNo);
                // For Address
                string Address = fun.CompAdd(CompId);
                cryRpt.SetParameterValue("Address", Address);
                CrystalReportViewer1.ReportSource = cryRpt;
                Session[Key] = cryRpt;
            }
            catch (Exception ex)
            {
            }
            finally
            {
                dt.Clear();
                dt.Dispose();
                DS.Clear();
                DS.Dispose();
                Min.Clear();
                Min.Dispose();
                con.Close();
                con.Dispose();
            }

           
        }

        else
        {
            ReportDocument doc = (ReportDocument)Session[Key];
            CrystalReportViewer1.ReportSource = doc;
        }
    }


    protected void Page_Load(object sender, EventArgs e)
    {
        cryRpt = new ReportDocument();  
    }

    protected void Page_UnLoad(object sender, EventArgs e)
    {
        this.CrystalReportViewer1.Dispose();
        this.CrystalReportViewer1 = null;
        cryRpt.Close();
        cryRpt.Dispose();
        GC.Collect();
    }
    protected void BtnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("MaterialIssueNote_MIN_Print.aspx?ModId=9&SubModId=41");
    }
    
}
