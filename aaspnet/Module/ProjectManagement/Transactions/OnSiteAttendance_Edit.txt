﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;

using System.Web.UI.HtmlControls;

using System.Xml.Linq;

using MKB.TimePicker;

public partial class Module_ProjectManagement_Transactions_OnSiteAttendance_Edit : System.Web.UI.Page
{
    string connStr = "";
    SqlConnection con;
    clsFunctions fun = new clsFunctions();
    int CompId = 0;
    int FinYearId = 0;
    string sId = "";
    string CDate = "";
    string CTime = "";
    
    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            connStr = fun.Connection();
            con = new SqlConnection(connStr);
            Label3.Text = textChequeDate.Text;
            CompId = Convert.ToInt32(Session["compid"]);
            FinYearId = Convert.ToInt32(Session["finyear"]);
            sId = Session["username"].ToString();
            CDate = fun.getCurrDate();
            CTime = fun.getCurrTime();
            textChequeDate.Attributes.Add("readonly", "readonly");

            if (!IsPostBack)
            {
                this.FillGrid(0);
            }
        }
        catch (Exception ex)
        { 
        
        }
    }

    public void FillGrid(int value)
    {
        try
        {
            DateTime SysDt = Convert.ToDateTime(String.Format("{0:MM-dd-yyyy}", DateTime.Now));
           
            DateTime EnteredDt = Convert.ToDateTime(String.Format("{0:MM-dd-yyyy}", fun.FromDateMDY(textChequeDate.Text)));            
            TimeSpan diff=EnteredDt.Subtract(SysDt);

            int x = 0;

            if (diff.Days == -1 || diff.Days>=0)
            {
                x = 1;
            }
            else if (diff.Days == -2)                
            {                
                DateTime addin = EnteredDt.AddDays(1);

                if (addin.DayOfWeek.ToString()=="Sunday")
                {
                    x = 1;
                }
            }

            if (x == 1)
            {
                SqlDataAdapter da = new SqlDataAdapter("GetOnSiteEmp", con);
                da.SelectCommand.CommandType = CommandType.StoredProcedure;
                da.SelectCommand.Parameters.Add(new SqlParameter("@BG", SqlDbType.VarChar));
                da.SelectCommand.Parameters.Add(new SqlParameter("@CompId", SqlDbType.VarChar));
                da.SelectCommand.Parameters.Add(new SqlParameter("@OnSiteDate", SqlDbType.VarChar));
                da.SelectCommand.Parameters["@BG"].Value = value;
                da.SelectCommand.Parameters["@CompId"].Value = CompId;
                da.SelectCommand.Parameters["@OnSiteDate"].Value = fun.FromDate(textChequeDate.Text).ToString();
                DataSet DSitem = new DataSet();
                da.Fill(DSitem);
                GridView4.DataSource = DSitem;
                GridView4.DataBind();
            }
            else
            {
                string myStringVariable = string.Empty;
                myStringVariable = "Invalid date!";
                ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + myStringVariable + "');", true);
            }
        }
        catch (Exception ex)
        { 
        
        }
    }
    protected void btnProceed_Click(object sender, EventArgs e)
    {
        this.FillGrid(Convert.ToInt32(drpGroupF.SelectedValue));
    }
    protected void GridView4_RowEditing(object sender, GridViewEditEventArgs e)
    {
        try
        {
            GridView4.EditIndex = e.NewEditIndex;
            this.FillGrid(Convert.ToInt32(drpGroupF.SelectedValue));
            int index = GridView4.EditIndex;
            GridViewRow grv = GridView4.Rows[index];
            int id = Convert.ToInt32(((Label)grv.FindControl("lblIdE")).Text);
            string Shift = ((Label)grv.FindControl("lblShiftE")).Text;
            string Status = ((Label)grv.FindControl("lblStatusE")).Text;
            if (Shift == "Day")
            {
                ((RadioButtonList)grv.FindControl("RadioButtonShift")).SelectedValue = "0";

            }
            else
            {
                ((RadioButtonList)grv.FindControl("RadioButtonShift")).SelectedValue = "1";

            }
           
            if (Status == "Present")
            {
                ((RadioButtonList)grv.FindControl("RadioButtonStatus")).SelectedValue = "0";

            }
            else
            {
                ((RadioButtonList)grv.FindControl("RadioButtonStatus")).SelectedValue = "1";

            }

            fun.TimeSelectorDatabase(((Label)grv.FindControl("lblFrmTimeE")).Text, ((TimeSelector)grv.FindControl("FTime")));

            if (((Label)grv.FindControl("lblToTimeE")).Text != string.Empty)
            {
                fun.TimeSelectorDatabase(((Label)grv.FindControl("lblToTimeE")).Text, ((TimeSelector)grv.FindControl("TTime")));
            }

            
        }
        catch (Exception ex)
        { }

    }
    protected void GridView4_RowCancelingEdit(object sender, GridViewCancelEditEventArgs e)
    {
        try
        {
            GridView4.EditIndex = -1;
            this.FillGrid(Convert.ToInt32(drpGroupF.SelectedValue));
        }
        catch (Exception er) { }
    }
    protected void GridView4_RowUpdating(object sender, GridViewUpdateEventArgs e)
    {
        try
        {
            int index1 = GridView4.EditIndex;
            GridViewRow grv = GridView4.Rows[index1];
            int id1 = Convert.ToInt32(GridView4.DataKeys[e.RowIndex].Value);


            if (((TextBox)grv.FindControl("txtOnsite")).Text != "")
            {

                int rad = Convert.ToInt32(((RadioButtonList)grv.FindControl("RadioButtonShift")).SelectedValue);
                int Status = Convert.ToInt32(((RadioButtonList)grv.FindControl("RadioButtonStatus")).SelectedValue);
                string onSite = ((TextBox)grv.FindControl("txtOnsite")).Text;
                string FromTime = ((TimeSelector)grv.FindControl("FTime")).Hour.ToString("D2") + ":" + ((TimeSelector)grv.FindControl("FTime")).Minute.ToString("D2") + ":" + ((TimeSelector)grv.FindControl("FTime")).Second.ToString("D2") + " " + ((TimeSelector)grv.FindControl("FTime")).AmPm.ToString();
                string ToTime = ((TimeSelector)grv.FindControl("TTime")).Hour.ToString("D2") + ":" + ((TimeSelector)grv.FindControl("TTime")).Minute.ToString("D2") + ":" + ((TimeSelector)grv.FindControl("TTime")).Second.ToString("D2") + " " + ((TimeSelector)grv.FindControl("TTime")).AmPm.ToString();
                if (Convert.ToDateTime(FromTime).TimeOfDay < Convert.ToDateTime(ToTime).TimeOfDay)
                {
                    string StrUpdate = fun.update("tblOnSiteAttendance_Master", "Shift='" + rad + "', Status='" + Status + "',UpSysDate='" + CDate + "',UpSessionId='" + sId + "',UpSysTime='" + CTime + "',Onsite='" + onSite + "',FromTime='" + FromTime + "',ToTime='"+ToTime+"'", "Id='" + id1 + "'");
                    SqlCommand cmd11 = new SqlCommand(StrUpdate, con);
                    con.Open();
                    cmd11.ExecuteNonQuery();
                    con.Close();
                    GridView4.EditIndex = -1;
                    this.FillGrid(Convert.ToInt32(drpGroupF.SelectedValue));                    
                }
                else
                {
                    string mystring = string.Empty;
                    mystring = "From Time Should Be Less than To Time.";
                    ClientScript.RegisterStartupScript(this.GetType(), "myalert", "alert('" + mystring + "');", true);
                }
            }
        }
        catch (Exception ex)
        {
        }
        finally
        {
            con.Close();

        }
    }
    protected void GridView4_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        GridView4.PageIndex = e.NewPageIndex;
        this.FillGrid(Convert.ToInt32(drpGroupF.SelectedValue));
    }

}
