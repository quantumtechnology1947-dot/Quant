﻿using System;

{
    clsFunctions fun = new clsFunctions();
    string sId = "";
    int CompId = 0;
    string PRNo = "";
    string MId = "";
    string str = "";
    SqlConnection con ;
    protected void Page_Load(object sender, EventArgs e)
    {
        try
        {
            PRNo = Request.QueryString["PRNo"];
            sId = Session["username"].ToString();
            CompId = Convert.ToInt32(Session["compid"]);           
            MId = Request.QueryString["Id"].ToString();
            str = fun.Connection();
            con = new SqlConnection(str);
            if (!IsPostBack)
            {
                this.LoadData();
            }
        }
        catch (Exception ep)
        {

        }
    }

    public void disableDelete()
    {
       
        foreach (GridViewRow grv in GridView3.Rows)
        {
            string Id = ((Label)grv.FindControl("lblid")).Text;
            string StrSql2 = fun.select("tblMM_PO_Details.Id", "tblMM_PO_Details,tblMM_PO_Master", "tblMM_PO_Details.PRNo='" + PRNo + "' AND tblMM_PO_Master.PONo=tblMM_PO_Details.PONo AND tblMM_PO_Master.CompId='" + CompId + "' AND tblMM_PO_Details.PRId='" + Id + "' AND tblMM_PO_Details.MId=tblMM_PO_Master.Id");

            SqlCommand cmdsupId2 = new SqlCommand(StrSql2, con);
            SqlDataAdapter dasupId2 = new SqlDataAdapter(cmdsupId2);
            DataSet DSSql2 = new DataSet();
            dasupId2.Fill(DSSql2);

            if (DSSql2.Tables[0].Rows.Count > 0)
            {
                ((LinkButton)grv.FindControl("Btndel")).Visible = false;
                ((Label)grv.FindControl("lblDel")).Visible = true;
            }
            else
            {
                ((LinkButton)grv.FindControl("Btndel")).Visible = true;
                ((Label)grv.FindControl("lblDel")).Visible = false;
            }
        }
    }


    public void LoadData()
    {
        
        con.Open();

        try
        {
            DataTable dt = new DataTable();
            dt.Columns.Add(new System.Data.DataColumn("PRNo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Description", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("AccHead", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Qty", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("Rate", typeof(double)));
            dt.Columns.Add(new System.Data.DataColumn("Remarks", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("Supplier", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Discount", typeof(double)));
            DataRow dr;
            string sql = fun.select("tblMM_PR_Master.PRNo,tblMM_PR_Details.ItemId,tblMM_PR_Details.Discount,tblMM_PR_Details.SupplierId,tblMM_PR_Details.AHId,tblMM_PR_Master.WONo,tblMM_PR_Details.Qty,tblMM_PR_Details.Rate,tblMM_PR_Details.Remarks,tblMM_PR_Details.Id", "tblMM_PR_Master,tblMM_PR_Details", "tblMM_PR_Master.PRNo='" + PRNo + "' AND tblMM_PR_Master.PRNo=tblMM_PR_Details.PRNo AND tblMM_PR_Master.Id=tblMM_PR_Details.MId AND tblMM_PR_Details.MId='" + MId + "' AND tblMM_PR_Master.CompId='" + CompId + "' order by Id desc");

            SqlCommand cmd = new SqlCommand(sql, con);
            SqlDataAdapter da = new SqlDataAdapter(cmd);
            DataSet DS = new DataSet();
            da.Fill(DS);

            for (int p = 0; p < DS.Tables[0].Rows.Count; p++)
            {
                dr = dt.NewRow();
                dr[0] = DS.Tables[0].Rows[p]["PRNo"].ToString();

                // For Item Code
                string sql1 = fun.select("ItemCode,ManfDesc,UOMBasic", "tblDG_Item_Master", "Id='" + Convert.ToInt32(DS.Tables[0].Rows[p]["ItemId"]) + "' AND CompId='" + CompId + "'");

                SqlCommand cmd1 = new SqlCommand(sql1, con);
                SqlDataAdapter da1 = new SqlDataAdapter(cmd1);
                DataSet DS1 = new DataSet();
                da1.Fill(DS1);
                if (DS1.Tables[0].Rows.Count>0)
                {
                    dr[1] = fun.GetItemCode_PartNo(CompId, Convert.ToInt32(DS.Tables[0].Rows[p]["ItemId"].ToString()));
                dr[2] = DS1.Tables[0].Rows[0]["ManfDesc"].ToString();
                string sql4 = fun.select("Symbol", "Unit_Master", "Id ='" + DS1.Tables[0].Rows[0]["UOMBasic"].ToString() + "'");
                SqlCommand cmd4 = new SqlCommand(sql4, con);
                SqlDataAdapter da4 = new SqlDataAdapter(cmd4);
                DataSet DS4 = new DataSet();
                da4.Fill(DS4);
                dr[3] = DS4.Tables[0].Rows[0]["Symbol"].ToString();
                }

                // For A/c Head
                string sql3 = fun.select("Symbol AS Head", "AccHead", "Id='" + Convert.ToInt32(DS.Tables[0].Rows[p]["AHId"]) + "' ");
                SqlCommand cmd3 = new SqlCommand(sql3, con);
                SqlDataAdapter da3 = new SqlDataAdapter(cmd3);
                DataSet DS3 = new DataSet();
                da3.Fill(DS3);
                if (DS3.Tables[0].Rows.Count > 0)
                {
                    dr[4] = DS3.Tables[0].Rows[0]["Head"].ToString();
                }
                dr[5] = DS.Tables[0].Rows[p]["WONo"].ToString();
                dr[6] = Convert.ToDouble(decimal.Parse(DS.Tables[0].Rows[p]["Qty"].ToString()).ToString("N3"));
                dr[7] = Convert.ToDouble(decimal.Parse(DS.Tables[0].Rows[p]["Rate"].ToString()).ToString("N2"));
                dr[8] = DS.Tables[0].Rows[p]["Remarks"].ToString();
                dr[9] = DS.Tables[0].Rows[p]["Id"].ToString();
                // For Supplier Name
                string sql2 = fun.select("tblMM_Supplier_master.SupplierName", "tblMM_Supplier_master,tblMM_PR_Details", "tblMM_Supplier_master.SupplierId ='" + DS.Tables[0].Rows[p]["SupplierId"] + "' AND tblMM_Supplier_master.SupplierId=tblMM_PR_Details.SupplierId AND tblMM_PR_Details.PRNo='" + DS.Tables[0].Rows[p]["PRNo"] + "' AND tblMM_Supplier_master.CompId='" + CompId + "'");

                SqlCommand cmd2 = new SqlCommand(sql2, con);
                SqlDataAdapter da2 = new SqlDataAdapter(cmd2);
                DataSet DS2 = new DataSet();
                da2.Fill(DS2);
                if (DS2.Tables[0].Rows.Count > 0)
                {
                    dr[10] = DS2.Tables[0].Rows[0]["SupplierName"].ToString();
                }

                dr[11] = Convert.ToDouble(decimal.Parse(DS.Tables[0].Rows[p]["Discount"].ToString()).ToString("N2"));
                dt.Rows.Add(dr);
                dt.AcceptChanges();
            }

            
            GridView3.DataSource = dt;
            GridView3.DataBind();
            this.disableDelete();
        }
        catch (Exception ex) { }

    }

    protected void GridView3_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        GridView3.PageIndex = e.NewPageIndex;
        this.LoadData();
    }
    protected void GridView3_RowCommand(object sender, GridViewCommandEventArgs e)
    {
        
       
        if (e.CommandName == "del")
        {
            GridViewRow row = (GridViewRow)(((LinkButton)e.CommandSource).NamingContainer);
           try
            {
                
                int id = Convert.ToInt32(((Label)row.FindControl("lblid")).Text);
                string sql = fun.delete("tblMM_PR_Details", "PRNo='" + PRNo + "' AND Id='" + id + "'");
                SqlCommand cmd = new SqlCommand(sql, con);
                con.Open();
                cmd.ExecuteNonQuery();
                con.Close();
                string Strdel = fun.select("*", "tblMM_PR_Details", "PRNo='" + PRNo + "' AND MId='" + MId + "'");
                SqlCommand cmddel = new SqlCommand(Strdel, con);
                SqlDataAdapter dadel = new SqlDataAdapter(cmddel);
                DataSet DSdel = new DataSet();
                dadel.Fill(DSdel);

                if (DSdel.Tables[0].Rows.Count == 0)
                {
                    string sqlDelM = fun.delete("tblMM_PR_Master", "PRNo='" + PRNo + "' AND CompId='" + CompId + "' AND Id='" + MId + "'");
                    SqlCommand cmdDelM = new SqlCommand(sqlDelM, con);
                    con.Open();
                    cmdDelM.ExecuteNonQuery();
                    con.Close();
                    Response.Redirect("PR_Delete.aspx?ModId=6&SubModId=34");
                }
                else
                {

                    this.LoadData();
                    //Page.Response.Redirect(Page.Request.Url.ToString(), true);
                }
            }
           catch (Exception ex) { }
           finally
            {
               

            }

        }

    }
    protected void btnCancel_Click(object sender, EventArgs e)
    {
        Response.Redirect("PR_Delete.aspx?ModId=6&SubModId=34");
    }
