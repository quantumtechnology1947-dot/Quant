﻿using System;
using System.Collections;
using System.Configuration;
using System.Data;
using System.Linq;
using System.Web;
using System.Web.Security;
using System.Web.UI;
using System.Web.UI.HtmlControls;
using System.Web.UI.WebControls;
using System.Web.UI.WebControls.WebParts;
using System.Xml.Linq;
using System.Data.SqlClient;

public partial class Module_Design_Transactions_TPL_Amd : System.Web.UI.Page
{
    clsFunctions fun = new clsFunctions();
    int CompId = 0;
    string sId = "";
    int FinYearId = 0;
    string WONo = "";
    string ItemId = "";
    string TPLId = "";
    protected void Page_Load(object sender, EventArgs e)
    {
        string connStr = fun.Connection();
        SqlConnection con = new SqlConnection(connStr);
        try
        {
            sId = Session["username"].ToString();
            CompId = Convert.ToInt32(Session["compid"]);
            FinYearId = Convert.ToInt32(Session["finyear"]);
            ItemId = Request.QueryString["ItemId"].ToString();
            WONo = Request.QueryString["WONo"].ToString();
            TPLId = Request.QueryString["Id"].ToString();
            lblWONo.Text = WONo;
            if (!IsPostBack)
            {
                this.BindDataCust();
            }

        }
        catch (Exception ex)
        {
        }
        finally
        {
            con.Close();
        }
    }



    public void BindDataCust()
    {
        try
        {

            string connStr = fun.Connection();
            SqlConnection con = new SqlConnection(connStr);
            DataTable dt = new DataTable();
            con.Open();

            string strAmd = fun.select("*", "tblDG_TPL_Amd", "FinYearId<='" + FinYearId + "' And CompId='" + CompId + "' And TPLId='" + TPLId + "' And WONo='" + WONo + "' And ItemId='" + ItemId + "'");
            SqlCommand CmdAmd = new SqlCommand(strAmd, con);
            SqlDataAdapter DaAmd = new SqlDataAdapter(CmdAmd);
            DataSet DSAmd = new DataSet();

            DaAmd.Fill(DSAmd);


            dt.Columns.Add(new System.Data.DataColumn("SysDate", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("SysTime", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("WONo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("AmdNo", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ItemCode", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("ManfDesc", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("UOM", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Qty", typeof(string)));
            dt.Columns.Add(new System.Data.DataColumn("Id", typeof(int)));
            dt.Columns.Add(new System.Data.DataColumn("AmdBy", typeof(string)));

            DataRow dr;
            for (int i = 0; i < DSAmd.Tables[0].Rows.Count; i++)
            {

                dr = dt.NewRow();

                if (DSAmd.Tables[0].Rows.Count > 0)
                {
                    string StrCust = fun.select("ItemCode,UOMBasic,ManfDesc", "tblDG_Item_Master", " FinYearId<='" + FinYearId + "'AND CompId='" + CompId + "'  AND  Id='" + DSAmd.Tables[0].Rows[i]["ItemId"].ToString() + "'");
                    SqlCommand cmdCust = new SqlCommand(StrCust, con);
                    SqlDataAdapter daCust = new SqlDataAdapter(cmdCust);
                    DataSet DSCust = new DataSet();
                    daCust.Fill(DSCust);



                    if (DSCust.Tables[0].Rows.Count > 0)
                    {



                        dr[4] = DSCust.Tables[0].Rows[0]["ItemCode"].ToString();


                        if (DSAmd.Tables[0].Rows[i]["Description"].ToString() != string.Empty)
                        {
                            dr[5] = DSAmd.Tables[0].Rows[i]["Description"].ToString();

                        }
                        else
                        {
                            dr[5] = DSCust.Tables[0].Rows[0]["ManfDesc"].ToString();
                        }


                        if (DSAmd.Tables[0].Rows[i]["UOM"].ToString() != string.Empty)
                        {

                            string StrSymbol = fun.select("Symbol", "Unit_Master", "  Id='" + DSCust.Tables[0].Rows[0]["UOMBasic"].ToString() + "'");
                            SqlCommand cmdSymbol = new SqlCommand(StrSymbol, con);
                            SqlDataAdapter daSymbol = new SqlDataAdapter(cmdSymbol);
                            DataSet DSSymbol = new DataSet();
                            daSymbol.Fill(DSSymbol);
                            if (DSSymbol.Tables[0].Rows.Count > 0)
                            {
                                dr[6] = DSSymbol.Tables[0].Rows[0]["Symbol"].ToString();
                            }




                        }
                        else
                        {
                            string StrSymbol = fun.select("Symbol", "Unit_Master", "  Id='" + DSAmd.Tables[0].Rows[i]["UOM"].ToString() + "'");
                            SqlCommand cmdSymbol = new SqlCommand(StrSymbol, con);
                            SqlDataAdapter daSymbol = new SqlDataAdapter(cmdSymbol);
                            DataSet DSSymbol = new DataSet();
                            daSymbol.Fill(DSSymbol);
                            if (DSSymbol.Tables[0].Rows.Count > 0)
                            {
                                dr[6] = DSSymbol.Tables[0].Rows[0]["Symbol"].ToString();
                            }

                        }

                    }


                    string strEmp = fun.select("Title+'.'+EmployeeName As EmpLoyeeName", "tblHR_OfficeStaff", "FinYearId<='" + FinYearId + "'And CompId='" + CompId + "' AND EmpId='" + DSAmd.Tables[0].Rows[i]["SessionId"] + "'");

                    SqlCommand cmdEmp = new SqlCommand(strEmp, con);
                    SqlDataAdapter daEmp = new SqlDataAdapter(cmdEmp);
                    DataSet DSEmp = new DataSet();
                    daEmp.Fill(DSEmp);

                    dr[0] = fun.FromDateDMY(DSAmd.Tables[0].Rows[i]["SysDate"].ToString());
                    dr[1] = DSAmd.Tables[0].Rows[i]["SysTime"].ToString();
                    dr[2] = DSAmd.Tables[0].Rows[i]["WONo"].ToString();
                    dr[3] = DSAmd.Tables[0].Rows[i]["AmdNo"].ToString();


                    dr[4] = DSAmd.Tables[0].Rows[i]["WONo"].ToString();

                    dr[7] = DSAmd.Tables[0].Rows[i]["Qty"].ToString();
                    dr[8] = DSAmd.Tables[0].Rows[i]["Qty"].ToString();
                    if (DSEmp.Tables[0].Rows.Count > 0)
                    {
                        dr[9] = DSEmp.Tables[0].Rows[0]["EmployeeName"].ToString();
                    }


                    dt.Rows.Add(dr);
                    dt.AcceptChanges();

                }
            }
            GridView2.DataSource = dt;
            GridView2.DataBind();
            con.Close();
        }

        catch (Exception ex)
        { }
    }


    protected void GridView2_PageIndexChanging(object sender, GridViewPageEventArgs e)
    {
        GridView2.EditIndex = e.NewPageIndex;
        this.BindDataCust();
    }
    protected void Button2_Click(object sender, EventArgs e)
    {
        Response.Redirect("TPL_Design_WO_TreeView_Edit.aspx?WONo=" + WONo + "&ModId=3&SubModId=23");
    }
}
