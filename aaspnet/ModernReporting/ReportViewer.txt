using System;
using System.Data;
using System.Web.UI;

/// <summary>
/// Standalone Report Viewer Page - Displays reports in popup windows
/// </summary>
public partial class ModernReporting_ReportViewer : System.Web.UI.Page
{
    private DataTable _reportData;
    private string _reportTitle;
    
    protected void Page_Load(object sender, EventArgs e)
    {
        if (!IsPostBack)
        {
            LoadReportFromSession();
        }
    }

    /// <summary>
    /// Load report content from session
    /// </summary>
    private void LoadReportFromSession()
    {
        try
        {
            string sessionKey = Request.QueryString["key"];
            
            if (!string.IsNullOrEmpty(sessionKey))
            {
                object reportContent = Session[sessionKey];
                
                if (reportContent != null)
                {
                    if (reportContent is string)
                    {
                        // Direct HTML content
                        litReportContent.Text = reportContent.ToString();
                    }
                    else if (reportContent is DataTable)
                    {
                        // DataTable - generate HTML
                        _reportData = (DataTable)reportContent;
                        _reportTitle = Session[sessionKey + "_Title"]?.ToString() ?? "Report";
                        
                        string htmlContent = ModernReportHelper.ReportGenerator.GenerateHTMLReport(
                            _reportData, _reportTitle);
                        litReportContent.Text = htmlContent;
                    }
                    
                    // Clean up session
                    Session.Remove(sessionKey);
                    Session.Remove(sessionKey + "_Title");
                }
                else
                {
                    ShowError("Report session has expired. Please generate the report again.");
                }
            }
            else
            {
                ShowError("Invalid report request. No session key provided.");
            }
        }
        catch (Exception ex)
        {
            ShowError("Error loading report: " + ex.Message);
        }
    }

    /// <summary>
    /// Show error message
    /// </summary>
    private void ShowError(string message)
    {
        pnlError.Visible = true;
        lblErrorMessage.Text = message;
        btnDownloadPDF.Visible = false;
    }

    /// <summary>
    /// Download report as PDF
    /// </summary>
    protected void btnDownloadPDF_Click(object sender, EventArgs e)
    {
        try
        {
            if (_reportData != null && _reportData.Rows.Count > 0)
            {
                byte[] pdfContent = ModernReportHelper.ReportGenerator.GeneratePDFReport(
                    _reportData, _reportTitle);

                string fileName = (_reportTitle ?? "Report").Replace(" ", "_") + "_" + DateTime.Now.ToString("yyyyMMdd_HHmmss");
                ModernReportHelper.ReportResponse.DownloadPDF(pdfContent, fileName);
            }
            else
            {
                // Try to generate PDF from current HTML content
                ClientScript.RegisterStartupScript(this.GetType(), "PDFAlert", 
                    "alert('PDF generation requires original data. Please use the PDF option from the main report page.');", true);
            }
        }
        catch (Exception ex)
        {
            ClientScript.RegisterStartupScript(this.GetType(), "ErrorAlert", 
                "alert('Error generating PDF: " + ex.Message.Replace("'", "\\'") + "');", true);
        }
    }
}
